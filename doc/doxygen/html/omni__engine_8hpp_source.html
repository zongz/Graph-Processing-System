38  /**
39  * \ingroup engines
40  *
41  * \brief The omni engine encapsulates all the GraphLab engines
42  * allowing the user to select which engine to use at runtime.
43  *
44  * The actual engine type is set as a string argument to the
45  * constructor of the omni_engine. Forexample:
46  *
47  * \code
48  * std::string exec_model = "synchronous";
49  * // do something to determine the exec_model (possibly command
50  * // line processing)
51  * // Create the engine
52  * graphlab::omni_engine<pagerank_vprog> engine(dc, graph, opts, exec_model);
53  * \endcode
54  *
55  * The specific engine type can be overriden by command line
56  * arguments (engine_opts="type=<type>"):
57  *
58  * \code
59  * graphlab::omni_engine<pagerank_vprog> engine(dc, graph, opts, "synchronous");
60  * \endcode
61  *
62  * then calling the progam with the command line options:
63  *
64  \verbatim
65  %> mpiexec -n 16 ./pagerank --engine_opts="type=synchronous"
66  \endverbatim
67  *
68  * The currently supproted types are:
69  *
70  * \li "synchronous" or "sync": uses the synchronous engine
71  * (\ref synchronous_engine)
72  * \li "asynchronous" or "async": uses the asynchronous engine
73  * (\ref async_consistent_engine)
74 *
75  * \see graphlab::synchronous_engine
76  * \see graphlab::async_consistent_engine
77  *
78  */

161  /**
162  * \brief Construct an omni engine for a given graph with the
163  * default_engine_type unless the engine options contain an
164  * alternative type.
165  *
166  * \param [in] dc a distributed control object that is used to
167  * connect this engine with it's counter parts on other machines.
168  * \param [in,out] graph the graph object that this engine will
169  * transform.
170  * \param [in] options the command line options which are used to
171  * configure the engine. Note that the engine option "type" can
172  * be used to select the engine to use (synchronous or
173  * asynchronous).
174  * \param [in] default_engine_type The user must specify what
175  * engine type to use if no command line option is given.
176  */
177  omni_engine(distributed_control& dc, graph_type& graph,
178  const std::string& default_engine_type,
179  const graphlab_options& options = graphlab_options()) :
180  engine_ptr(NULL) {
181  graphlab_options new_options = options;
182  std::string engine_type = default_engine_type;
183  options_map& engine_options = new_options.get_engine_args();
184  if(engine_options.get_option("type", engine_type)) {
185  // the engine option was set so use it instead
186  // clear from the options map
187  engine_options.options.erase("type");
188  }
189  // Process the engine types
190  if(engine_type == "sync" || engine_type == "synchronous") {
191  logstream(LOG_INFO) << "Using the Synchronous engine." << std::endl;
192  engine_ptr = new synchronous_engine_type(dc, graph, new_options);
193  } else if(engine_type == "async" || engine_type == "asynchronous") {
194  logstream(LOG_INFO) << "Using the Asynchronous engine." << std::endl;
195  engine_ptr = new async_consistent_engine_type(dc, graph, new_options);
196  } else {
197  logstream(LOG_FATAL) << "Invalid engine type: " << engine_type << std::endl;
198  }
199  } // end of constructor
200
