99  /** Add a message to the set returning false if a message is already
100  present. */
101  bool add(const size_t idx,
102  const value_type& val,
103  double* message_priority = NULL) {
104  double priority;
105  size_t lockidx = get_lock_idx(idx);
106  lock_array[lockidx].lock();
107  bool ret = message_vector[idx].add(val, priority);
108  joincounter[lockidx] += !ret;
109  addcounter[lockidx]++;
110  lock_array[lockidx].unlock();
111  if (message_priority) (*message_priority) = priority;
112  return ret;
113  }

115  /** Returns the current message stored at idx and
116  * clears the message.
117  * Returns true on success and false if there is no message
118  * stored at the index.
119  */
120  bool get(const size_t idx,
121  value_type& ret_val) {
122  bool has_val = false;
123  size_t lockidx = get_lock_idx(idx);
124  lock_array[lockidx].lock();
125  if (!message_vector[idx].empty) {
126  ret_val = message_vector[idx].value;
127  message_vector[idx].clear();
128  has_val = true;
129  }
130  lock_array[lockidx].unlock();
