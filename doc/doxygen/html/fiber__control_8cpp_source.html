45 fiber_control::fiber_control(size_t nworkers,
46  size_t affinity_base)
47  :nworkers(nworkers),
48  affinity_base(affinity_base),
49  stop_workers(false),
50  flsdeleter(NULL) {
51  // initialize the thread local storage keys
52  if (!tls_created) {
53  pthread_key_create(&tlskey, fiber_control::tls_deleter);
54  tls_created = true;
55  }
56 
57  // set up the queues.
58  schedule.resize(nworkers);
59  for (size_t i = 0;i < nworkers; ++i) {
60  schedule[i].waiting = false;
61  schedule[i].nwaiting = 0;
62  schedule[i].affinity_queue = new inplace_lf_queue2<fiber>;
63  schedule[i].priority_queue = new inplace_lf_queue2<fiber>;
64  schedule[i].popped_affinity_queue = NULL;
65  schedule[i].popped_priority_queue = NULL;
66  }
67  // launch the workers
68  for (size_t i = 0;i < nworkers; ++i) {
69  workers.launch(boost::bind(&fiber_control::worker_init, this, i),
70  affinity_base + i);
71  }
72 }

143 fiber_control::fiber* fiber_control::try_pop_queue(inplace_lf_queue2<fiber>& lfqueue,
144  fiber*& popped_queue) {
145  fiber_control::fiber* ret = NULL;
146  // if there is stuff in the popped queue, pop it.
147  if (popped_queue == NULL) {
148  popped_queue = lfqueue.dequeue_all();
149  }
150 
151  if (popped_queue != NULL) {
152  ret = popped_queue;
153  do {
154  popped_queue = ret->next;
155  asm volatile("pause\n": : :"memory");
156  } while(popped_queue == NULL);
157  // we have reached the end of the queue. clear the popped queue
158  // and return
159  if (popped_queue == lfqueue.end_of_dequeue_list()) {
160  popped_queue = NULL;
161  }
162  }
163  return ret;
164 }

196 void fiber_control::worker_init(size_t workerid) {
197  /*
198  * This is the "root" stack for each worker.
199  * When there are active user threads associated with this worker,
200  * it will switch directly between the fibers.
201  * But, when the worker has no other fiber to run, it will return to this
202  * stack and and wait in a condition variable
203  */
204  // create a root context
205  create_tls_ptr();
206  // set up the tls structure
207  tls* t = get_tls_ptr();
208  t->prev_fiber = NULL;
209  t->cur_fiber = NULL;
210  t->garbage = NULL;
211  t->workerid = workerid;
212  t->parent = this;
213 
214  schedule[workerid].waiting = true;
215  schedule[workerid].active_lock.lock();
216  while(!stop_workers) {
217  // get a fiber to run
218  fiber* next_fib = t->parent->active_queue_remove(workerid);
219  if (next_fib != NULL) {
220  // if there is a fiber. yield to it
221  schedule[workerid].active_lock.unlock();
222  schedule[workerid].waiting = false;
223  active_workers.inc();
224  yield_to(next_fib);
225  if (flush_timer.current_time() > 0.0001 && flush_lock.try_lock()) {
226  distributed_control* dc = distributed_control::get_instance();
227  if (dc) dc->flush_soon();
228  flush_timer.start();
229  flush_lock.unlock();
230  }
231  active_workers.dec();
232  schedule[workerid].waiting = true;
233  schedule[workerid].active_lock.lock();
234  } else {
235  // if there is no fiber. wait.
236  schedule[workerid].active_cond.wait(schedule[workerid].active_lock);
237  }
238  }
239  schedule[workerid].active_lock.unlock();
240 }

247 void fiber_control::trampoline(intptr_t _args) {
248  // we may have launched to here by switching in from another fiber.
249  // we will need to clean up the previous fiber
250  tls* t = get_tls_ptr();
251  if (t->prev_fiber) t->parent->reschedule_fiber(t->workerid, t->prev_fiber);
252  t->prev_fiber = NULL;
253 
254  trampoline_args* args = reinterpret_cast<trampoline_args*>(_args);
255  try {
256  args->fn();
257  } catch (...) {
258  }
259  delete args;
260  fiber_control::exit();
261 }

263 size_t fiber_control::launch(boost::function<void(void)> fn,
264  size_t stacksize,
265  affinity_type affinity) {
266  ASSERT_GT(affinity.popcount(), 0);
267  size_t b = 0;
268  ASSERT_TRUE(affinity.first_bit(b));
269  // make sure there is always a worker I can work on
270  ASSERT_LT(b, nworkers);
271 
272  // allocate a stack
273  fiber* fib = new fiber;
274  fib->parent = this;
275  fib->stack = malloc(stacksize);
276  fib->id = fiber_id_counter.inc();
277  foreach(size_t b, affinity) {
278  if (b < nworkers) fib->affinity_array.push_back((unsigned char)b);
279  else break;
280  }
281  ASSERT_GT(fib->affinity_array.size(), 0);
282  fib->affinity = affinity;
283  //VALGRIND_STACK_REGISTER(fib->stack, (char*)fib->stack + stacksize);
284  fib->fls = NULL;
285  fib->next = NULL;
286  fib->deschedule_lock = NULL;
287  fib->terminate = false;
288  fib->descheduled = false;
289  fib->scheduleable = true;
290  // construct the initial context
291  trampoline_args* args = new trampoline_args;
292  args->fn = fn;
293  fib->initial_trampoline_args = (intptr_t)(args);
294  // stack grows downwards.
295  fib->context = boost::context::make_fcontext((char*)fib->stack + stacksize,
296  stacksize,
297  trampoline);
298  fibers_active.inc();
299 
300  // find a place to put the thread
301  size_t choice = pick_fiber_worker(fib);
302  active_queue_insert_tail(choice, fib);
303  return reinterpret_cast<size_t>(fib);
304 }

306 size_t fiber_control::pick_fiber_worker(fiber* fib) {
307  // first try to use the original worker if possible
308  size_t choice = get_worker_id();
309  if (choice == (size_t)(-1) || fib->affinity.get(choice) == 0) {
310  //choice rejected, pick randomly from the available choices
311  // if there is only one affinity option, return it
312  if (fib->affinity_array.size() == 1) {
313  choice = fib->affinity_array[0];
314  } else {
315  size_t ra = graphlab::random::fast_uniform<size_t>(0,fib->affinity_array.size() - 1);
316  std::swap(fib->affinity_array[ra], fib->affinity_array[0]);
317  choice = fib->affinity_array[0];
318  }
319  }
320  return choice;
321 }

323 void fiber_control::yield_to(fiber* next_fib) {
324  // the core scheduling logic
325  tls* t = get_tls_ptr();
326 
327 // if (next_fib) {
328 // if (t->cur_fiber) {
329 // printf("%ld: yield to: %ld from %ld\n", get_worker_id(), next_fib->id, t->cur_fiber->id);
330 // } else {
331 // printf("%ld: yield to: %ld\n", get_worker_id(), next_fib->id);
332 // }
333 // }
334  if (next_fib != NULL) {
335  // reset the priority flag
336  next_fib->priority = false;
337  // current fiber moves to previous
338  // next fiber move to current
339  t->prev_fiber = t->cur_fiber;
340  t->cur_fiber = next_fib;
341  if (t->prev_fiber != NULL) {
342  // context switch to fib outside the lock
343  boost::context::jump_fcontext(t->prev_fiber->context,
344  t->cur_fiber->context,
345  t->cur_fiber->initial_trampoline_args);
346  } else {
347  boost::context::jump_fcontext(&t->base_context,
348  t->cur_fiber->context,
349  t->cur_fiber->initial_trampoline_args);
350  }
351  } else {
352  // ok. there isn't anything to schedule to
353  // am I meant to be terminated? or descheduled?
354  if (t->cur_fiber &&
355  (t->cur_fiber->terminate || t->cur_fiber->descheduled) ) {
356  // yup. killing current fiber
357  // context switch back to basecontext which will
358  // do the cleanup
359  //
360  // current fiber moves to previous
361  // next fiber (base context) move to current
362  // (as identifibed by cur_fiber = NULL)
363  t->prev_fiber = t->cur_fiber;
364  t->cur_fiber = NULL;
365  boost::context::jump_fcontext(t->prev_fiber->context,
366  &t->base_context,
367  0);
368  } else {
369  // nothing to do, and not terminating...
370  // then don't yield!
371  return;
372  }
373  }
374  // reread the tls pointer because we may have woken up in a different thread
375  t = get_tls_ptr();
376  // reschedule the previous fiber
377  if (t->prev_fiber) reschedule_fiber(t->workerid, t->prev_fiber);
378  t->prev_fiber = NULL;
379 
380  // if distributed_controller alive
381  distributed_control* dc = distributed_control::get_instance();
382  if (dc && t->workerid < dc->num_handler_threads()) {
383  dc->handle_incoming_calls(t->workerid, dc->num_handler_threads());
384  }
385 }

387 void fiber_control::reschedule_fiber(size_t workerid, fiber* fib) {
388  fib->lock.lock();
389  if (!fib->terminate && !fib->descheduled) {
390  fib->lock.unlock();
391  // we reschedule it
392  // Re-lock the queue
393  //printf("%ld: Reinserting %ld\n", get_worker_id(), fib->id);
394  if (!fib->priority) active_queue_insert_tail(workerid, fib);
395  else active_queue_insert_head(workerid, fib);
396  } else if (fib->descheduled) {
397  // unflag descheduled and unset scheduleable
398  fib->descheduled = false;
399  fib->scheduleable = false;
400  if (fib->deschedule_lock) pthread_mutex_unlock(fib->deschedule_lock);
401  fib->deschedule_lock = NULL;
402  //printf("%ld: Descheduling complete %ld\n", get_worker_id(), fib->id);
403  fib->lock.unlock();
404  } else if (fib->terminate) {
405  fib->lock.unlock();
406  // previous fiber is dead. destroy it
407  free(fib->stack);
408  //VALGRIND_STACK_DEREGISTER(fib->stack);
409  // delete the fiber local storage if any
410  if (fib->fls && flsdeleter) flsdeleter(fib->fls);
411  delete fib;
412  // if we are out of threads, signal the join
413  if (fibers_active.dec() == 0) {
414  join_lock.lock();
415  join_cond.signal();
416  join_lock.unlock();
417  }
418  } else {
419  // impossible condition
420  assert(false);
421  }
422 }

495 void fiber_control::schedule_tid(size_t tid, bool priority) {
496  fiber* fib = reinterpret_cast<fiber*>(tid);
497  fib->lock.lock();
498  // we MUST get here only after the thread was completely descheduled
499  // or no deschedule operation has happened yet.
500  assert(fib->descheduled == false);
501  fib->descheduled = false;
502  if (fib->scheduleable == false) {
503  // if this thread was descheduled completely. Reschedule it.
504  //printf("%ld: Scheduling requested %ld\n", get_worker_id(), fib->id);
505  fib->scheduleable = true;
506  fib->priority = priority;
507  fib->lock.unlock();
508  size_t choice = fib->parent->pick_fiber_worker(fib);
509  fib->parent->reschedule_fiber(choice, fib);
510  } else {
511  //printf("%ld: Scheduling requested of running thread %ld\n", get_worker_id(), fib->id);
512  fib->lock.unlock();
513  }
514 }
