38  /**
39  * A compact key-value(s) data structure using Compressed Sparse Row format.
40  * The key has type size_t and can be assolicated with multiple values of valuetype.
41  *
42  * The core operation of is querying the list of values associated with the query key
43  * and returns the begin and end iterators via <code>begin(id)</code> and <code>end(id)</code>.
44  *
45  * Also, this class supports insert (and batch insert) values associated with any key.
46  */
47  template<typename valuetype, typename sizetype=size_t,
48  uint32_t blocksize=(4096-20)/(4*sizeof(valuetype))> // the block size makes the block fit in a memory page

124  /// Insert a range of values to a given key
125  template <typename idtype, typename InputIterator>
126  void insert (const idtype& key, InputIterator first, InputIterator last) {
127  if (last-first == 0) {
128  return;
129  }
130  // iterator to the insertion position
131  iterator ins_iter = end(key);
132 
133  // begin_ins_iter and end_ins_iterator point to
134  // defines the range of the new inserted element.
135  std::pair<iterator,iterator> iter_pair = values.insert(ins_iter, first, last);
136  iterator begin_ins_iter = iter_pair.first;
137  iterator end_ins_iter = iter_pair.second;
138 
139  // add blocks for new key
140  while (key >= num_keys()) {
141  value_ptrs.push_back(begin_ins_iter);
142  }
143 
144  // Update pointers.
145  // value_ptrs[key] = begin_ins_iter;
146  // ASSERT_TRUE(begin_ins_iter == ins_iter);
147 
148  // Update pointers to the right of ins_iter.
149  // Base case: the pointer of ins_iter is mapped to end_ins_iter.
150  uint32_t oldoffset = ins_iter.get_offset();
151  iterator newiter = end_ins_iter;
152  for (size_t scan = key+1; scan < num_keys(); ++scan) {
153  if (value_ptrs[scan].get_blockptr() == ins_iter.get_blockptr()) {
154  while (oldoffset != value_ptrs[scan].get_offset()) {
155  ++oldoffset;
156  ++newiter;
157  }
158  value_ptrs[scan] = newiter;
159  } else {
160  break;
161  }
162  }
163  }

212  std::vector<sizetype> valueptr_vec(num_keys(), 0);
213  for (size_t i = 1;i < num_keys(); ++i) {
214  const_iterator begin_iter = begin(i - 1);
215  const_iterator end_iter = end(i - 1);
216  sizetype length = begin_iter.pdistance_to(end_iter);
217  valueptr_vec[i] = valueptr_vec[i - 1] + length;
218  }

220  std::vector<valuetype> out;

248  /**
249  * Initialize the internal member with input key_vec and value_vec.
250  * value_vec will be compactly wrapped into a block_linked_list
251  * and key_vec will be converted into an array of iterators (pointers)
252  * to the values in the block_linked_list.
253  */
254  template<typename idtype>
255  void init(const std::vector<idtype>& id_vec,
256  const std::vector<valuetype>& value_vec) {
257  ASSERT_EQ(id_vec.size(), value_vec.size());
258  std::vector<sizetype> permute_index;
259 
260  // Build index for id -> value
261  // Prefix of the counting array equals to the begin index for each id
262  std::vector<sizetype> prefix;
263  counting_sort(id_vec, permute_index, &prefix);
264 
265  // Fill in the value vector
266  typedef boost::permutation_iterator<
267  typename std::vector<valuetype>::const_iterator,
268  typename std::vector<sizetype>::const_iterator> permute_iterator;
269  permute_iterator _begin = boost::make_permutation_iterator(value_vec.begin(), permute_index.begin());
270  permute_iterator _end = boost::make_permutation_iterator(value_vec.end(), permute_index.end());
271  values.assign(_begin, _end);
272 
273  // Fill in the key vector
274  sizevec2ptrvec(prefix, value_ptrs);
275 
276  // Build the index pointers
277 #ifdef DEBUG_CSR
278  for (size_t i = 0; i < permute_index.size(); ++i)
279  std::cerr << permute_index[i] << " ";
280  std::cerr << std::endl;
281 
282  for (size_t i = 0; i < value_ptrs.size(); ++i)
283  std::cerr << prefix[i] << " ";
284  std::cerr << std::endl;
285 
286  for (permute_iterator it = _begin; it != _end; ++it) {
287  std::cerr << *it << " ";
288  }
289  std::cerr << std::endl;
290 
291  for (size_t i = 0; i < num_keys(); ++i) {
292  std::cerr << i << ": ";
293  iterator it = begin(i);
294  while (it != end(i)) {
295  std::cerr << *it << " ";
296  ++it;
297  }
298  std::cerr << std::endl;
299  }
300  std::cerr << std::endl;
301 #endif
302  }

304  // Convert integer pointers into block_linked_list::value_iterator
305  // Assuming all blocks are fully packed.
306  void sizevec2ptrvec (const std::vector<sizetype>& ptrs,
307  std::vector<iterator>& out) {
308  ASSERT_EQ(out.size(), 0);
309  out.reserve(ptrs.size());
310 
311  // for efficiency, we advance pointers based on the previous value
312  // because block_linked_list is mostly forward_traversal.
313  iterator it = values.begin();
314  sizetype prev = 0;
315  for (size_t i = 0; i < ptrs.size(); ++i) {
316  sizetype cur = ptrs[i];
317  it += (cur-prev);
318  out.push_back(it);
319  prev = cur;
320  }
321  }
