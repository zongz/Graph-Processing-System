61  /**
62  * \ingroup engines
63  *
64  * \brief The asynchronous consistent engine executed vertex programs
65  * asynchronously and can ensure mutual exclusion such that adjacent vertices
66  * are never executed simultaneously. The default mode is "factorized"
67  * consistency in which only individual gathers/applys/
68  * scatters are guaranteed to be consistent, but this can be strengthened to
69  * provide full mutual exclusion.
70  *
71  *
72  * \tparam VertexProgram
73  * The user defined vertex program type which should implement the
74  * \ref graphlab::ivertex_program interface.
75  *
76  * ### Execution Semantics
77  *
78  * On start() the \ref graphlab::ivertex_program::init function is invoked
79  * on all vertex programs in parallel to initialize the vertex program,
80  * vertex data, and possibly signal vertices.
81  *
82  * After which, the engine spawns a collection of threads where each thread
83  * individually performs the following tasks:
84  * \li Extract a message from the scheduler.
85  * \li Perform distributed lock acquisition on the vertex which is supposed
86  * to receive the message. The lock system enforces that no neighboring
87  * vertex is executing at the same time. The implementation is based
88  * on the Chandy-Misra solution to the dining philosophers problem.
89  * (Chandy, K.M.; Misra, J. (1984). The Drinking Philosophers Problem.
90  * ACM Trans. Program. Lang. Syst)
91  * \li Once lock acquisition is complete,
92  * \ref graphlab::ivertex_program::init is called on the vertex
93  * program. As an optimization, any messages sent to this vertex
94  * before completion of lock acquisition is merged into original message
95  * extracted from the scheduler.
96  * \li Execute the gather on the vertex program by invoking
97  * the user defined \ref graphlab::ivertex_program::gather function
98  * on the edge direction returned by the
99  * \ref graphlab::ivertex_program::gather_edges function. The gather
100  * functions can modify edge data but cannot modify the vertex
101  * program or vertex data and can be executed on multiple
102  * edges in parallel.
103  * * \li Execute the apply function on the vertex-program by
104  * invoking the user defined \ref graphlab::ivertex_program::apply
105  * function passing the sum of the gather functions. If \ref
106  * graphlab::ivertex_program::gather_edges returns no edges then
107  * the default gather value is passed to apply. The apply function
108  * can modify the vertex program and vertex data.
109  * \li Execute the scatter on the vertex program by invoking
110  * the user defined \ref graphlab::ivertex_program::scatter function
111  * on the edge direction returned by the
112  * \ref graphlab::ivertex_program::scatter_edges function. The scatter
113  * functions can modify edge data but cannot modify the vertex
114  * program or vertex data and can be executed on multiple
115  * edges in parallel.
116  * \li Release all locks acquired in the lock acquisition stage,
117  * and repeat until the scheduler is empty.
118  *
119  * The engine threads multiplexes the above procedure through a secondary
120  * internal queue, allowing an arbitrary large number of vertices to
121  * begin processing at the same time.
122  *
123  * ### Construction
124  *
125  * The asynchronous consistent engine is constructed by passing in a
126  * \ref graphlab::distributed_control object which manages coordination
127  * between engine threads and a \ref graphlab::distributed_graph object
128  * which is the graph on which the engine should be run. The graph should
129  * already be populated and cannot change after the engine is constructed.
130  * In the distributed setting all program instances (running on each machine)
131  * should construct an instance of the engine at the same time.
132  *
133  * Computation is initiated by signaling vertices using either
134  * \ref graphlab::async_consistent_engine::signal or
135  * \ref graphlab::async_consistent_engine::signal_all. In either case all
136  * machines should invoke signal or signal all at the same time. Finally,
137  * computation is initiated by calling the
138  * \ref graphlab::async_consistent_engine::start function.
139  *
140  * ### Example Usage
141  *
142  * The following is a simple example demonstrating how to use the engine:
143  * \code
144  * #include <graphlab.hpp>
145  *
146  * struct vertex_data {
147  * // code
148  * };
149  * struct edge_data {
150  * // code
151  * };
152  * typedef graphlab::distributed_graph<vertex_data, edge_data> graph_type;
153  * typedef float gather_type;
154  * struct pagerank_vprog :
155  * public graphlab::ivertex_program<graph_type, gather_type> {
156  * // code
157  * };
158  *
159  * int main(int argc, char** argv) {
160  * // Initialize control plain using mpi
161  * graphlab::mpi_tools::init(argc, argv);
162  * graphlab::distributed_control dc;
163  * // Parse command line options
164  * graphlab::command_line_options clopts("PageRank algorithm.");
165  * std::string graph_dir;
166  * clopts.attach_option("graph", &graph_dir, graph_dir,
167  * "The graph file.");
168  * if(!clopts.parse(argc, argv)) {
169  * std::cout << "Error in parsing arguments." << std::endl;
170  * return EXIT_FAILURE;
171  * }
172  * graph_type graph(dc, clopts);
173  * graph.load_structure(graph_dir, "tsv");
174  * graph.finalize();
175  * std::cout << "#vertices: " << graph.num_vertices()
176  * << " #edges:" << graph.num_edges() << std::endl;
177  * graphlab::async_consistent_engine<pagerank_vprog> engine(dc, graph, clopts);
178  * engine.signal_all();
179  * engine.start();
180  * std::cout << "Runtime: " << engine.elapsed_seconds();
181  * graphlab::mpi_tools::finalize();
182  * }
183  * \endcode
184  *
185  * \see graphlab::omni_engine
186  * \see graphlab::synchronous_engine
187  *
188  * <a name=engineopts>Engine Options</a>
189  * =========================
190  * The asynchronous engine supports several engine options which can
191  * be set as command line arguments using \c --engine_opts :
192  *
193  * \li \b timeout (default: infinity) Maximum time in seconds the engine will
194  * run for. The actual runtime may be marginally greater as the engine
195  * waits for all threads and processes to flush all active tasks before
196  * returning.
197  * \li \b factorized (default: true) Set to true to weaken the consistency
198  * model to factorized consistency where only individual gather/apply/scatter
199  * calls are guaranteed to be locally consistent. Can produce massive
200  * increases in throughput at a consistency penalty.
201  * \li \b nfibers (default: 10000) Number of fibers to use
202  * \li \b stacksize (default: 16384) Stacksize of each fiber.
203  */

462  /**
463  * \internal
464  * Configures the engine with the provided options.
465  * The number of threads to create are read from
466  * opts::get_ncpus(). The scheduler to construct is read from
467  * graphlab_options::get_scheduler_type(). The default scheduler
468  * is the queued_fifo scheduler. For details on the scheduler types
469  * \see scheduler_types
470  */
471  void set_options(const graphlab_options& opts) {
472  rmi.barrier();
473  ncpus = opts.get_ncpus();
474  ASSERT_GT(ncpus, 0);
475  aggregation_lock.resize(opts.get_ncpus());
476  aggregation_queue.resize(opts.get_ncpus());
477  std::vector<std::string> keys = opts.get_engine_args().get_option_keys();
478  foreach(std::string opt, keys) {
479  if (opt == "timeout") {
480  opts.get_engine_args().get_option("timeout", timed_termination);
481  if (rmi.procid() == 0)
482  logstream(LOG_EMPH) << "Engine Option: timeout = " << timed_termination << std::endl;
483  } else if (opt == "factorized") {
484  opts.get_engine_args().get_option("factorized", factorized_consistency);
485  if (rmi.procid() == 0)
486  logstream(LOG_EMPH) << "Engine Option: factorized = " << factorized_consistency << std::endl;
487  } else if (opt == "nfibers") {
488  opts.get_engine_args().get_option("nfibers", nfibers);
489  if (rmi.procid() == 0)
490  logstream(LOG_EMPH) << "Engine Option: nfibers = " << nfibers << std::endl;
491  } else if (opt == "track_task_time") {
492  opts.get_engine_args().get_option("track_task_time", track_task_time);
493  if (rmi.procid() == 0)
494  logstream(LOG_EMPH) << "Engine Option: track_task_time = " << track_task_time<< std::endl;
495  }else if (opt == "stacksize") {
496  opts.get_engine_args().get_option("stacksize", stacksize);
497  if (rmi.procid() == 0)
498  logstream(LOG_EMPH) << "Engine Option: stacksize= " << stacksize << std::endl;
499  } else if (opt == "use_cache") {
500  opts.get_engine_args().get_option("use_cache", use_cache);
501  if (rmi.procid() == 0)
502  logstream(LOG_EMPH) << "Engine Option: use_cache = " << use_cache << std::endl;
503  } else {
504  logstream(LOG_FATAL) << "Unexpected Engine Option: " << opt << std::endl;
505  }
506  }
507  opts_copy = opts;
508  // set a default scheduler if none
509  if (opts_copy.get_scheduler_type() == "") {
510  opts_copy.set_scheduler_type("queued_fifo");
511  }
512 
513  // construct scheduler passing in the copy of the options from set_options
514  scheduler_ptr = scheduler_factory::
515  new_scheduler(graph.num_local_vertices(),
516  opts_copy);
517  rmi.barrier();
518 
519  // create initial fork arrangement based on the alternate vid mapping
520  if (factorized_consistency == false) {
521  cmlocks = new distributed_chandy_misra<graph_type>(rmi.dc(), graph,
522  boost::bind(&engine_type::lock_ready, this, _1));
523 
524  }
525  else {
526  cmlocks = NULL;
527  }
528 
529  // construct the termination consensus object
530  consensus = new fiber_async_consensus(rmi.dc(), nfibers);
531  }

553  if (!factorized_consistency) {
554  cm_handles.resize(graph.num_local_vertices());
555  }
556  rmi.barrier();
557  }

612  /**
613  * \internal
614  * \brief Signals a vertex with an optional message
615  *
616  * Signals a vertex, and schedules it to be executed in the future.
617  * must be called on a vertex accessible by the current machine.
618  */
619  void internal_signal(const vertex_type& vtx,
620  const message_type& message = message_type()) {
621  if (force_stop) return;
622  if (started) {
623  const typename graph_type::vertex_record& rec = graph.l_get_vertex_record(vtx.local_id());
624  const procid_t owner = rec.owner;
625  if (endgame_mode) {
626  // fast signal. push to the remote machine immediately
627  if (owner != rmi.procid()) {
628  const vertex_id_type vid = rec.gvid;
629  rmi.remote_call(owner, &engine_type::rpc_signal, vid, message);
630  }
631  else {
632  double priority;
633  messages.add(vtx.local_id(), message, &priority);
634  scheduler_ptr->schedule(vtx.local_id(), priority);
635  consensus->cancel();
636  }
637  }
638  else {
639 
640  double priority;
641  messages.add(vtx.local_id(), message, &priority);
642  scheduler_ptr->schedule(vtx.local_id(), priority);
643  consensus->cancel();
644  }
645  }
646  else {
647  double priority;
648  messages.add(vtx.local_id(), message, &priority);
649  scheduler_ptr->schedule(vtx.local_id(), priority);
650  consensus->cancel();
651  }
652  } // end of schedule

818  lvid_type& sched_lvid,
819  message_type &msg) {

839  }
840  bool ret = consensus->end_done_critical_section(threadid);
841  if (ret == false) {

851  << "\tCancelled by Scheduler Task" << std::endl;
852  consensus->cancel_critical_section(threadid);
853  has_sched_msg = true;
854  return false;
855  }

874  conditional_gather_type perform_gather(vertex_id_type vid,
875  vertex_program_type& vprog_) {
876  vertex_program_type vprog = vprog_;
877  lvid_type lvid = graph.local_vid(vid);
878  local_vertex_type local_vertex(graph.l_vertex(lvid));
879  vertex_type vertex(local_vertex);
880  context_type context(*this, graph);
881  edge_dir_type gather_dir = vprog.gather_edges(context, vertex);
882  conditional_gather_type accum;
883 
884  //check against the cache
885  if( use_cache && has_cache.get(lvid) ) {
886  accum.set(gather_cache[lvid]);
887  return accum;
888  }
889  // do in edges
890  if(gather_dir == IN_EDGES || gather_dir == ALL_EDGES) {
891  foreach(local_edge_type local_edge, local_vertex.in_edges()) {
892  edge_type edge(local_edge);
893  lvid_type a = edge.source().local_id(), b = edge.target().local_id();
894  vertexlocks[std::min(a,b)].lock();
895  vertexlocks[std::max(a,b)].lock();
896  accum += vprog.gather(context, vertex, edge);
897  vertexlocks[a].unlock();
898  vertexlocks[b].unlock();
899  }
900  }
901  // do out edges
902  if(gather_dir == OUT_EDGES || gather_dir == ALL_EDGES) {
903  foreach(local_edge_type local_edge, local_vertex.out_edges()) {
904  edge_type edge(local_edge);
905  lvid_type a = edge.source().local_id(), b = edge.target().local_id();
906  vertexlocks[std::min(a,b)].lock();
907  vertexlocks[std::max(a,b)].lock();
908  accum += vprog.gather(context, vertex, edge);
909  vertexlocks[a].unlock();
910  vertexlocks[b].unlock();
911  }
912  }
913  if (use_cache) {
914  gather_cache[lvid] = accum.value; has_cache.set_bit(lvid);
915  }
916  return accum;
917  }

1004  /**
1005  * \internal
1006  * Called when the scheduler returns a vertex to run.
1007  * If this function is called with vertex locks acquired, prelocked
1008  * should be true. Otherwise it should be false.
1009  */
1010  void eval_sched_task(const lvid_type lvid,
1011  const message_type& msg) {
1012  const typename graph_type::vertex_record& rec = graph.l_get_vertex_record(lvid);
1013  vertex_id_type vid = rec.gvid;
1014  char task_time_data[sizeof(timer)];
1015  timer* task_time;
1016  if (track_task_time) {
1017  // placement new to create the timer
1018  task_time = reinterpret_cast<timer*>(task_time_data);
1019  new (task_time) timer();
1020  }
1021  // if this is another machine's forward it
1022  if (rec.owner != rmi.procid()) {
1023  rmi.remote_call(rec.owner, &engine_type::rpc_signal, vid, msg);
1024  return;
1025  }
1026  // I have to run this myself
1027 
1028  if (!get_exclusive_access_to_vertex(lvid, msg)) return;
1029 
1030  /**************************************************************************/
1031  /* Acquire Locks */
1032  /**************************************************************************/
1033  if (!factorized_consistency) {
1034  // begin lock acquisition
1035  cm_handles[lvid] = new vertex_fiber_cm_handle;
1036  cm_handles[lvid]->philosopher_ready = false;
1037  cm_handles[lvid]->fiber_handle = fiber_control::get_tid();
1038  cmlocks->make_philosopher_hungry(lvid);
1039  cm_handles[lvid]->lock.lock();
1040  while (!cm_handles[lvid]->philosopher_ready) {
1041  fiber_control::deschedule_self(&(cm_handles[lvid]->lock.m_mut));
1042  cm_handles[lvid]->lock.lock();
1043  }
1044  cm_handles[lvid]->lock.unlock();
1045  }
1046 
1047  /**************************************************************************/
1048  /* Begin Program */
1049  /**************************************************************************/
1050  context_type context(*this, graph);
1051  vertex_program_type vprog = vertex_program_type();
1052  local_vertex_type local_vertex(graph.l_vertex(lvid));
1053  vertex_type vertex(local_vertex);
1054 
1055  /**************************************************************************/
1056  /* init phase */
1057  /**************************************************************************/
1058  vprog.init(context, vertex, msg);
1059 
1060  /**************************************************************************/
1061  /* Gather Phase */
1062  /**************************************************************************/
1063  conditional_gather_type gather_result;
1064  std::vector<request_future<conditional_gather_type> > gather_futures;
1065  foreach(procid_t mirror, local_vertex.mirrors()) {
1066  gather_futures.push_back(
1067  object_fiber_remote_request(rmi,
1068  mirror,
1069  &async_consistent_engine::perform_gather,
1070  vid,
1071  vprog));
1072  }
1073  gather_result += perform_gather(vid, vprog);
1074 
1075  for(size_t i = 0;i < gather_futures.size(); ++i) {
1076  gather_result += gather_futures[i]();
1077  }
1078 
1079  /**************************************************************************/
1080  /* apply phase */
1081  /**************************************************************************/
1082  vertexlocks[lvid].lock();
1083  vprog.apply(context, vertex, gather_result.value);
1084  vertexlocks[lvid].unlock();
1085 
1086 
1087  /**************************************************************************/
1088  /* scatter phase */
1089  /**************************************************************************/
1090 
1091  // should I wait for the scatter? nah... but in case you want to
1092  // the code is commented below
1093  /*foreach(procid_t mirror, local_vertex.mirrors()) {
1094  rmi.remote_call(mirror,
1095  &async_consistent_engine::perform_scatter,
1096  vid,
1097  vprog,
1098  local_vertex.data());
1099  }*/
1100 
1101  std::vector<request_future<void> > scatter_futures;
1102  foreach(procid_t mirror, local_vertex.mirrors()) {
1103  scatter_futures.push_back(
1104  object_fiber_remote_request(rmi,
1105  mirror,
1106  &async_consistent_engine::perform_scatter,
1107  vid,
1108  vprog,
1109  local_vertex.data()));
1110  }
1111  perform_scatter_local(lvid, vprog);
1112  for(size_t i = 0;i < scatter_futures.size(); ++i)
1113  scatter_futures[i]();
1114 
1115  /************************************************************************/
1116  /* Release Locks */
1117  /************************************************************************/
1118  // the scatter is used to release the chandy misra
1119  // here I cleanup
1120  if (!factorized_consistency) {
1121  delete cm_handles[lvid];
1122  cm_handles[lvid] = NULL;
1123  }
1124  release_exclusive_access_to_vertex(lvid);
