54  static std::string add_line_breaks(const std::string &s, size_t numcols) {
55  size_t pos = 0;
56  std::string ret;
57  while(pos < s.length() - 1) {
58  size_t oldpos = pos;
59  pos = std::min(pos + numcols, s.length());
60 
61  size_t newpos = pos;
62  // search backward for a space if we are not at the end of the
63  // string
64  if (pos < s.length()) {
65  newpos = s.rfind(" ", pos);
66  }
67  // if we get back to the old position, or we fail to find a
68  // space, force the break
69  if (newpos == std::string::npos || newpos == oldpos) {
70  newpos = pos;
71  }
72  // break
73  ret = ret + trim(s.substr(oldpos, newpos - oldpos)) + "\n";
74  pos = newpos;
75  }
76  return ret;
77  }
