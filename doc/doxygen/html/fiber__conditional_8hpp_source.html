47  /// Waits on condition. The mutex must already be acquired. Caller
48  /// must be careful about spurious wakes.
49  inline void wait(const mutex& mut) const {
50  size_t tid = fiber_control::get_tid();
51  if (tid > 0) {
52  lock.lock();
53  fibers.push(tid);
54  lock.unlock();
55  fiber_control::deschedule_self(&mut.m_mut);
56  mut.lock();
57  } else {
58  int error = pthread_cond_wait(&m_cond, &mut.m_mut);
59  ASSERT_TRUE(!error);
60  }
61  }
62  /// Signals one waiting thread to wake up
63  inline void signal() const {
64  if (!fibers.empty()) {
65  lock.lock();
66  if (!fibers.empty()) {
67  size_t tid = fibers.front();
68  fibers.pop();
69  lock.unlock();
70  fiber_control::schedule_tid(tid);
71  return;
72  }
73  lock.unlock();
74  }
75  int error = pthread_cond_signal(&m_cond);
76  ASSERT_TRUE(!error);
77  }
78  /// Wakes up all waiting threads
79  inline void broadcast() const {
80  lock.lock();
