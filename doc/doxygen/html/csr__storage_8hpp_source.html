33  /**
34  * A compact key-value(s) data structure using Compressed Sparse Row format.
35  * The key has type size_t and can be assolicated with multiple values of valuetype.
36  * The core operation of is querying the list of values associated with the query key * and returns the begin and end iterators via <code>begin(id)</code>
37  * and <code>end(id)</code>.
38  */

59  template<typename idtype>
60  void init(const std::vector<idtype>& id_vec,
61  const std::vector<valuetype>& value_vec) {
62 
63  ASSERT_EQ(id_vec.size(), value_vec.size());
64 
65  std::vector<sizetype> permute_index;
66  // Build index for id -> value
67  // Prefix of the counting array equals to the begin index for each id
68  std::vector<sizetype>& prefix = value_ptrs;
69 
70  counting_sort(id_vec, permute_index, &prefix);
71 
72  values.reserve(value_vec.size());
73  values.resize(value_vec.size());
74  for (ssize_t i = 0; i < (ssize_t)value_vec.size(); ++i) {
75  values[i] = value_vec[permute_index[i]];
76  }
77 
78 #ifdef DEBUG_CSR
79  for (size_t i = 0; i < permute_index.size(); ++i)
80  std::cout << permute_index[i] << " ";
81  std::cout << std::endl;
82 
83  for (size_t i = 0; i < value_ptrs.size(); ++i)
84  std::cout << value_ptrs[i] << " ";
85  std::cout << std::endl;
86 
87  for (size_t i = 0; i < values.size(); ++i)
88  std::cout << values[i] << " ";
89  std::cout << std::endl;
90 
91  for (size_t i = 0; i < num_keys(); ++i) {
92  std::cout << i << ": ";
93  iterator it = begin(i);
94  while (it != end(i)) {
95  std::cout << *it << " ";
96  ++it;
97  }
98  std::cout << endl;
99  }
100  std::cout << std::endl;
101 #endif
102  }

104  /**
105  * Wrap the index vector and value vector into csr_storage.
106  * Check the property of the input vector.
107  * The input vector will be cleared.
108  */
109  void wrap(std::vector<sizetype>& valueptr_vec,
110  std::vector<valuetype>& value_vec) {
111  for (ssize_t i = 1; i < (ssize_t)valueptr_vec.size(); ++i) {
112  ASSERT_LE(valueptr_vec[i-1], valueptr_vec[i]);
113  ASSERT_LT(valueptr_vec[i], value_vec.size());
114  }
115  value_ptrs.swap(valueptr_vec);
116  values.swap(value_vec);
117  }
