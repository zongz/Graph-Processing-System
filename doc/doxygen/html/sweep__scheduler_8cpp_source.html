28 void sweep_scheduler::set_options(const graphlab_options& opts) {
29  std::vector<std::string> keys = opts.get_scheduler_args().get_option_keys();
30  bool max_iter_set = false;
31  foreach(std::string opt, keys) {
32  if (opt == "order") {
33  opts.get_scheduler_args().get_option("order", ordering);
34  ASSERT_TRUE(ordering == "random" || ordering == "ascending");
35  } else if (opt == "strict") {
36  opts.get_scheduler_args().get_option("strict", strict_round_robin);
37  } else if (opt == "max_iterations") {
38  opts.get_scheduler_args().get_option("max_iterations", max_iterations);
39  max_iter_set = true;
40  } else {
41  logstream(LOG_FATAL) << "Unexpected Scheduler Option: " << opt << std::endl;
42  }
43  }
44 
45  if (max_iter_set) {
46  ASSERT_MSG(strict_round_robin,
47  "sweep_scheduler: \"strict\" must be set with \"max_iteration\"");
48  }
49 }

51 sweep_scheduler::sweep_scheduler(size_t num_vertices,
52  const graphlab_options& opts) :
53  ncpus(opts.get_ncpus()),
54  num_vertices(num_vertices),
55  strict_round_robin(true),
56  max_iterations(std::numeric_limits<size_t>::max()),
57  vertex_is_scheduled(num_vertices) {
58  // initialize defaults
59  ASSERT_GE(opts.get_ncpus(), 1);
60  ordering = "random";
61  set_options(opts);
62 
63  if (ordering == "ascending") {
64  randomizer = 1;
65  } else if(ordering == "random") {
66  randomizer = 1500450271;
67  }
68 
69  if(strict_round_robin) {
70  logstream(LOG_INFO)
71  << "Using a strict round robin schedule." << std::endl;
72  // Max iterations only applies to strict round robin
73  if(max_iterations != std::numeric_limits<size_t>::max()) {
74  logstream(LOG_INFO)
75  << "Using maximum iterations: " << max_iterations << std::endl;
76  }
77  rr_index = 0;
78  } else {
79  // each cpu is responsible for its own subset of vertices
80  // Initialize the cpu2index counters
81  cpu2index.resize(ncpus);
82  for(size_t i = 0; i < cpu2index.size(); ++i) cpu2index[i] = i;
83  }
84  vertex_is_scheduled.resize(num_vertices);
85 } // end of constructor

98 sched_status::status_enum sweep_scheduler::get_next(const size_t cpuid,
99  lvid_type& ret_vid) {
100  const size_t max_fails = (num_vertices/ncpus) + 1;
101  // Check to see if max iterations have been achieved
102  if(strict_round_robin && (rr_index / num_vertices) >= max_iterations)
103  return sched_status::EMPTY;
104  // Loop through all vertices that are associated with this
105  // processor searching for a vertex with an active task
106  for(size_t idx = get_and_inc_index(cpuid), fails = 0;
107  fails <= max_fails; //
108  idx = get_and_inc_index(cpuid), ++fails) {
109  // It is possible that the get_and_inc_index could return an
110  // invalid index if the number of cpus exceeds the number of
111  // vertices. In This case we alwasy return empty
112  if(__builtin_expect(idx >= num_vertices, false)) return sched_status::EMPTY;
113  const lvid_type vid = (idx * randomizer) % num_vertices;
114  bool success = vertex_is_scheduled.clear_bit(vid);
115  while(success) { // Job found now decide whether to keep it
116  ret_vid = vid;
117  return sched_status::NEW_TASK;
118  }
119  } // end of for loop
120  return sched_status::EMPTY;
121 } // end of get_next
