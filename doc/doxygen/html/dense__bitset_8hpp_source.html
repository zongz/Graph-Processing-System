73  /** Resizes the current bitset to hold n bits.
74  Existing bits will not be changed. If the array size is increased,
75  the value of the new bits are undefined.
76 
77  \Warning When shirnking, the current implementation may still leave the
78  "deleted" bits in place which will mess up the popcount.
79  */
80  inline void resize(size_t n) {
81  len = n;
82  //need len bits
83  size_t prev_arrlen = arrlen;
84  arrlen = (n / (sizeof(size_t) * 8)) + (n % (sizeof(size_t) * 8) > 0);
85  array = (size_t*)realloc(array, sizeof(size_t) * arrlen);
86  // this zeros the remainder of the block after the last bit
87  fix_trailing_bits();
88  // if we grew, we need to zero all new blocks
89  if (arrlen > prev_arrlen) {
90  for (size_t i = prev_arrlen; i < arrlen; ++i) {
91  array[i] = 0;
92  }
93  }
94  }

156  /**
157  * \brief Transfers approximately b bits from another bitset to this bitset
158  *
159  * "Moves" at least b bits from the other bitset to this bitset
160  * starting from the given position.
161  * At return, b will contain the actual number of bits moved,
162  * and start will point to the end of the transfered region.
163  *
164  * Semantically what this accomplishes is something like:
165  *
166  * \code
167  * idx = start;
168  * if other.get_bit(idx) == false {
169  * idx = next true bit after idx in other (with loop around)
170  * }
171  * for(transferred = 0; transferred < b; transferred++) {
172  * other.clear_bit(idx);
173  * this->set_bit(idx);
174  * idx = next true bit after idx in other.
175  * if no more bits, return
176  * }
177  * \endcode
178  * However, the implementation here may transfer more than b bits.
179  * ( up to b + 2 * wordsize_in_bits )
180  */
181  inline void transfer_approximate_unsafe(dense_bitset& other,
182  size_t& start,
183  size_t& b) {
184  // must be identical in length
185  ASSERT_EQ(other.len, len);
186  ASSERT_EQ(other.arrlen, arrlen);
187  size_t arrpos, bitpos;
188  bit_to_pos(start, arrpos, bitpos);
189  size_t initial_arrpos = arrpos;
190  if (arrpos >= arrlen) arrpos = 0;
191  // ok. we will only look at arrpos
192  size_t transferred = 0;
193  while(transferred < b) {
194  if (other.array[arrpos] > 0) {
195  transferred += __builtin_popcountl(other.array[arrpos]);
196  array[arrpos] |= other.array[arrpos];
197  other.array[arrpos] = 0;
198  }
199  ++arrpos;
200  if (arrpos >= other.arrlen) arrpos = 0;
201  else if (arrpos == initial_arrpos) break;
202  }
203  start = 8 * sizeof(size_t) * arrpos;
204  b = transferred;
205  }

310  /** Returns true with b containing the position of the
311  first bit set to true.
312  If such a bit does not exist, this function returns false.
313  */
314  inline bool first_bit(size_t &b) const {
315  for (size_t i = 0; i < arrlen; ++i) {
316  if (array[i]) {
317  b = (size_t)(i * (sizeof(size_t) * 8)) + first_bit_in_block(array[i]);
318  return true;
319  }
320  }
321  return false;
322  }

325  /** Returns true with b containing the position of the
326  first bit set to false.
327  If such a bit does not exist, this function returns false.
328  */
329  inline bool first_zero_bit(size_t &b) const {
330  for (size_t i = 0; i < arrlen; ++i) {
331  if (~array[i]) {
332  b = (size_t)(i * (sizeof(size_t) * 8)) + first_bit_in_block(~array[i]);
333  return true;
334  }
335  }
336  return false;
337  }

339  /** Where b is a bit index, this function will return in b,
340  the position of the next bit set to true, and return true.
341  If all bits after b are false, this function returns false.
342  */
343  inline bool next_bit(size_t &b) const {
344  size_t arrpos, bitpos;
345  bit_to_pos(b, arrpos, bitpos);
346  //try to find the next bit in this block
347  bitpos = next_bit_in_block(bitpos, array[arrpos]);
348  if (bitpos != 0) {
349  b = (size_t)(arrpos * (sizeof(size_t) * 8)) + bitpos;
350  return true;
351  }
352  else {
353  // we have to loop through the rest of the array
354  for (size_t i = arrpos + 1; i < arrlen; ++i) {
355  if (array[i]) {
356  b = (size_t)(i * (sizeof(size_t) * 8)) + first_bit_in_block(array[i]);
357  return true;
358  }
359  }
360  }
361  return false;
362  }

464  // returns 0 on failure
465  inline size_t next_bit_in_block(const size_t& b, const size_t& block) const {
466  size_t belowselectedbit = size_t(-1) - (((size_t(1) << b) - 1)|(size_t(1)<<b));
467  size_t x = block & belowselectedbit ;
468  if (x == 0) return 0;
469  else return (size_t)__builtin_ctzl(x);
470  }

479  void fix_trailing_bits() {
480  // how many bits are in the last block
481  size_t lastbits = len % (8 * sizeof(size_t));
482  if (lastbits == 0) return;
483  array[arrlen - 1] &= ((size_t(1) << lastbits) - 1);
