210  // Program Steps ==========================================================
211  /**
212  * \brief Executes ncpus copies of a member function each with a
213  * unique consecutive id (thread id).
214  *
215  * This function is used by the main loop to execute each of the
216  * stages in parallel.
217  *
218  * The member function must have the type:
219  *
220  * \code
221  * void graph_gather_apply::member_fun(size_t threadid, const vertex_set& vset);
222  * \endcode
223  *
224  * This function runs an rmi barrier after termination
225  *
226  * @tparam the type of the member function.
227  * @param [in] member_fun the function to call.
228  */
229  template<typename MemberFunction>
230  void run_synchronous(MemberFunction member_fun, const vertex_set& vset) {
231  shared_lvid_counter = 0;
232  if (threads.size() <= 1) {
233  (this->*(member_fun))(0, vset);
234  }
235  else {
236  // launch the initialization threads
237  for(size_t i = 0; i < threads.size(); ++i) {
238  boost::function<void(void)> invoke = boost::bind(member_fun, this, i, vset);
239  threads.launch(invoke, i);
240  }
241  }
242  // Wait for all threads to finish
243  threads.join();
244  rmi.barrier();
245  } // end of run_synchronous

336  template<typename Graph, typename GatherType>
337  void graph_gather_apply<Graph,GatherType>::exec(const vertex_set& vset) {
338  if (vset.lazy && !vset.is_complete_set)
339  return;
340 
341  gather_accum.clear();
342  // Allocate vertex locks and vertex programs
343  vlocks.resize(graph.num_local_vertices());
344  // Allocate gather accumulators and accumulator bitset
345  gather_accum.resize(graph.num_local_vertices(), gather_type());
346  has_gather_accum.resize(graph.num_local_vertices());
347  has_gather_accum.clear();
348  rmi.barrier();
349 
350  // Execute gather operations-------------------------------------------
351  // Execute the gather operation for all vertices that are active
352  // in this minor-step (active-minorstep bit set).
353  // if (rmi.procid() == 0) std::cout << "Gathering..." << std::endl;
354  run_synchronous(&graph_gather_apply::execute_gathers, vset);
355 
356  // Execute the gather operation for all vertices that are active
357  // in this minor-step (active-minorstep bit set).
358  // if (rmi.procid() == 0) std::cout << "Gathering..." << std::endl;
359  run_synchronous(&graph_gather_apply::execute_scatters, vset);
360 
361 
362  // Execute Apply Operations -------------------------------------------
363  // Run the apply function on all active vertices
364  // if (rmi.procid() == 0) std::cout << "Applying..." << std::endl;
365  run_synchronous(&graph_gather_apply::execute_applys, vset);
366 
367  /**
368  * Post conditions:
369  * 1) any changes to the vertex data have been synchronized
370  * with all mirrors.
371  * 2) all gather accumulators have been cleared
372  */
373 
374  // Final barrier to ensure that all engines terminate at the same time
375  rmi.full_barrier();
376  } // end of start

379  template<typename Graph, typename GatherType>
380  void graph_gather_apply<Graph, GatherType>::
381  execute_gathers(const size_t thread_id, const vertex_set& vset) {
382  const bool TRY_TO_RECV = true;
383  const size_t TRY_RECV_MOD = 1000;
384  size_t vcount = 0;
385  timer ti;
386 
387  fixed_dense_bitset<8 * sizeof(size_t)> local_bitset;
388  while (1) {
389  // increment by a word at a time
390  lvid_type lvid_block_start =
391  shared_lvid_counter.inc_ret_last(8 * sizeof(size_t));
392  if (lvid_block_start >= graph.num_local_vertices()) break;
393 
394  local_bitset.clear();
395 
396  if (vset.lazy) {
397  ASSERT_TRUE(vset.is_complete_set);
398  local_bitset.fill();
399  } else {
400  // get the bit field from has_message
401  size_t lvid_bit_block = vset.localvset.containing_word(lvid_block_start);
402  if (lvid_bit_block == 0) continue;
403  // initialize a word sized bitfield
404  local_bitset.initialize_from_mem(&lvid_bit_block, sizeof(size_t));
405  }
406 
407  foreach(size_t lvid_block_offset, local_bitset) {
408  lvid_type lvid = lvid_block_start + lvid_block_offset;
409  if (lvid >= graph.num_local_vertices()) break;
410 
411  // std::cout << "proc " << rmi.procid() << " gather on lvid " << lvid << std::endl;
412  gather_type accum = gather_fun(lvid, graph);
413 
414  // If the accum contains a value for the local gather we put
415  // that estimate in the gather exchange.
416  sync_gather(lvid, accum, thread_id);
417 
418  // try to recv gathers if there are any in the buffer
419  if(++vcount % TRY_RECV_MOD == 0) recv_gathers(TRY_TO_RECV);
420  }
421  } // end of loop over vertices to compute gather accumulators
422  gather_exchange.partial_flush(thread_id);
423  // Finish sending and receiving all gather operations
424  thread_barrier.wait();
425  if(thread_id == 0) gather_exchange.flush();
426  thread_barrier.wait();
427  recv_gathers();
428  } // end of execute_gathers

521  // Data Synchronization ===================================================
522  template<typename Graph, typename GatherType>
523  void graph_gather_apply<Graph,GatherType>::
524  sync_gather(lvid_type lvid, const gather_type& accum, const size_t thread_id) {
525  if(graph.l_is_master(lvid)) {
526  vlocks[lvid].lock();
527  if(has_gather_accum.get(lvid)) {
528  gather_accum[lvid] += accum;
529  } else {
530  gather_accum[lvid] = accum;
531  has_gather_accum.set_bit(lvid);
532  }
533  vlocks[lvid].unlock();
534  } else {
535  const procid_t master = graph.l_master(lvid);
536  const vertex_id_type vid = graph.global_vid(lvid);
537  gather_exchange.send(master, std::make_pair(vid, accum), thread_id);
538  }
539  } // end of sync_gather

541  template<typename Graph, typename GatherType>
542  void graph_gather_apply<Graph, GatherType>::
543  scatter_gather(lvid_type lvid, const gather_type& accum, const size_t thread_id) {
544  ASSERT_TRUE(graph.l_is_master(lvid));
545  const vertex_id_type vid = graph.global_vid(lvid);
546  local_vertex_type vertex = graph.l_vertex(lvid);
547  foreach(const procid_t& mirror, vertex.mirrors()) {
548  gather_exchange.send(mirror, std::make_pair(vid, accum), thread_id);
549  }
550  } // end of sync_gather

552  template<typename Graph, typename GatherType>
553  void graph_gather_apply<Graph,GatherType>::
554  recv_gathers(const bool try_to_recv) {
555  procid_t procid(-1);
556  typename gather_exchange_type::buffer_type buffer;
557  while(gather_exchange.recv(procid, buffer, try_to_recv)) {
558  foreach(const vid_gather_pair_type& pair, buffer) {
559  ASSERT_TRUE(graph.vid2lvid.find(pair.first) != graph.vid2lvid.end());
560  const lvid_type lvid = graph.local_vid(pair.first);
561  const gather_type& accum = pair.second;
562  vlocks[lvid].lock();
563  if( has_gather_accum.get(lvid) ) {
564  gather_accum[lvid] += accum;
565  } else {
566  gather_accum[lvid] = accum;
567  has_gather_accum.set_bit(lvid);
568  }
569  vlocks[lvid].unlock();
570  }
571  }
572  } // end of recv_gather
