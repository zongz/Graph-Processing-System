55 bool init_param_from_zookeeper(dc_init_param& param) {
56  char* zk_hosts = getenv("ZK_SERVERS");
57  char* zk_jobname = getenv("ZK_JOBNAME");
58  char* zk_numnodes = getenv("ZK_NUMNODES");
59  if (zk_hosts == NULL || zk_jobname == NULL || zk_numnodes == NULL) {
60  return false;
61  }
62 
63  std::vector<std::string> zk_hosts_list = strsplit(zk_hosts, ",");
64 
65  // number of nodes to wait for
66  size_t numnodes = atoi(zk_numnodes);
67  ASSERT_GE(numnodes, 1);
68  logstream(LOG_EMPH) << "Using Zookeeper for Initialization. Waiting for "
69  << numnodes << " to join" << std::endl;
70 
71  // generate a unique identifier for this server
72 
73  std::pair<size_t, int> port_and_sock = get_free_tcp_port();
74  size_t port = port_and_sock.first;
75  int sock = port_and_sock.second;
76  std::string ipaddr = get_local_ip_as_str(true);
77  ipaddr = ipaddr + ":" + tostr(port);
78  logstream(LOG_INFO) << "Will Listen on: " << ipaddr << std::endl;
79 
80  // get an ip address
81  zookeeper::server_list server_list(zk_hosts_list,
82  zk_jobname,
83  ipaddr);
84 
85  // final server list goes here
86  std::vector<std::string> received_servers;
87  // locks to product the final server list
88  mutex lock;
89  conditional cond;
90 
91  // construct the watch to watch for changes on zookeeper
92  server_list.set_callback(boost::bind(zk_callback,
93  _1,
94  _2,
95  _3,
96  boost::ref(received_servers),
97  numnodes,
98  boost::ref(lock),
99  boost::ref(cond)));
100 
101  server_list.join("graphlab");
102 
103  lock.lock();
104  received_servers = server_list.watch_changes("graphlab");
105  // wait until I get all the servers
106  // TODO: add a timeout
107  while(received_servers.size() < numnodes) cond.wait(lock);
108  lock.unlock();
109 
110  // done!
111  // now make sure that everyone sees the server list in the same order
112 
113  ASSERT_EQ(received_servers.size(), numnodes);
114  std::sort(received_servers.begin(), received_servers.end());
115 
116  // now fill the parameter list
117  param.machines = received_servers;
118  param.curmachineid = std::find(received_servers.begin(), received_servers.end(),
119  ipaddr) - received_servers.begin();
120  ASSERT_LT(param.curmachineid, received_servers.size());
121  param.numhandlerthreads = RPC_DEFAULT_NUMHANDLERTHREADS;
122  param.commtype = RPC_DEFAULT_COMMTYPE;
123  param.initstring = param.initstring + std::string(" __sockhandle__=") + tostr(sock) + " ";
124  // detach from the server list
125  // now, this takes advantage of the Zookeeper feature that
126  // every machine sees all changes in the same order.
127  // i.e. At some point, everyone would have seen a complete server list.
128  // Once that happens, everyone can leave.
129  server_list.set_callback(NULL);
130  server_list.leave("graphlab");
131 
132  return true;
133 }
