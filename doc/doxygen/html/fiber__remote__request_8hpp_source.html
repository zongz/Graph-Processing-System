32 /**
33  * A implementation of the ireply_container interface
34  * that will wait for rpc requests, but if the request is issued from within
35  * a fiber, will deschedule the fiber.
36  */
37 struct fiber_reply_container: public dc_impl::ireply_container {
38  dc_impl::blob val;
39  mutex lock;
40  conditional cond;
41  // if wait is in a fiber, this will contain the ID of the fiber to wake up
42  // If 0, the wait is not in a fiber.
43  size_t waiting_tid;
44  // true when the blob is assigned
45  bool valready;
46 
47  fiber_reply_container():waiting_tid(0),valready(false) { }
48 
49  ~fiber_reply_container() {
50  val.free();
51  }
52 
53  void wait() {
54  if (fiber_control::in_fiber()) {
55  // if I am in a fiber, use the deschedule mechanism
56  lock.lock();
57  waiting_tid = fiber_control::get_tid();
58  while(!valready) {
59  // set the waiting tid value
60  // deschedule myself. This will deschedule the fiber
61  // and unlock the lock atomically
62  fiber_control::deschedule_self(&lock.m_mut);
63  // unlock the condition variable, this does not re-lock the lock
64  lock.lock();
65  }
66  lock.unlock();
67  } else {
68  // Otherwise use the condition variable
69  waiting_tid = 0;
70  lock.lock();
71  while(!valready) cond.wait(lock);
72  lock.unlock();
73  }
74  }
75 
76  void receive(procid_t source, dc_impl::blob b) {
77  lock.lock();
78  val = b;
79  valready = true;
80  if (waiting_tid) {
81  // it is a fiber. wake it up.
82  fiber_control::schedule_tid(waiting_tid);
83  } else {
84  // not in fiber. This is just a condition signal
85  cond.signal();
86  }
87  lock.unlock();
88  }
89  bool ready() const {
90  return valready;
91  }
92 
93  dc_impl::blob& get_blob() {
94  return val;
95  }
96 };

102 /**
103  * \brief Performs a nonblocking RPC call to the target machine
104  * to run the provided function pointer which has an expected return value.
105  *
106  * fiber_remote_request() calls the function "fn" on a target remote machine.
107  * Provided arguments are serialized and sent to the target.
108  * Therefore, all arguments are necessarily transmitted by value.
109  * If the target function has a return value, it is sent back to calling
110  * machine. fiber_remote_request() returns immediately a \ref
111  * graphlab::request_future object which will allow you wait for the return
112  * value.
113  *
114  * fiber_remote_request() has an identical interface to
115  * \ref graphlab::distributed_control::future_remote_request() , but has the
116  * additional capability that if a \ref graphlab::request_future::wait() is
117  * called on the request while within a fiber, it deschedules the fiber and
118  * context switches, returning only when the future is ready. This allows
119  * the future to be used from within a fiber.
120  *
121  * Since this function is not a member of the distributed_control class,
122  * it uses the function \ref distributed_control::get_instance() to obtain
123  * the last instance of the distribute_control class created. This should be
124  * sufficient for most use cases.
125  *
126  * \ref graphlab::object_fiber_remote_request is the version of this function
127  * for remotely calling class member functions.
128  *
129  * Example:
130  * \code
131  * // A print function is defined
132  * int add_one(int i) {
133  * return i + 1;
134  * }
135  *
136  * ... ...
137  * // call the add_one function on machine 1
138  * int i = 10;
139  * graphlab::request_future<int> ret = fiber_remote_request(1, add_one, i);
140  * // this is safe to do within a fiber as it will not halt other fibers.
141  * int result = ret();
142  * // result will be 11
143  * \endcode
144  *
145  * \see graphlab::distributed_control::remote_request
146  * graphlab::distributed_control::future_remote_request
147  * graphlab::object_fiber_remote_request
148  *
149  * \param targetmachine The ID of the machine to run the function on
150  * \param fn The function to run on the target machine. Must be a pointer to
151  * member function in the owning object.
152  * \param ... The arguments to send to Fn. Arguments must be serializable.
153  * and must be castable to the target types.
154  *
155  * \returns Returns a future templated around the same type as the return
156  * value of the called function
157  */
158  request_future<RetVal> fiber_remote_request(procid_t targetmachine, Fn fn, ...);

162 /**
163  * \brief Performs a nonblocking RPC call to the target machine
164  * to run the provided function pointer which has an expected return value.
165  *
166  * object_fiber_remote_request() calls the function "fn" on a target remote machine.
167  * Provided arguments are serialized and sent to the target.
168  * Therefore, all arguments are necessarily transmitted by value.
169  * If the target function has a return value, it is sent back to calling
170  * machine. object_fiber_remote_request() returns immediately a \ref
171  * graphlab::request_future object which will allow you wait for the return
172  * value.
173  *
174  * object_fiber_remote_request() has an identical interface to
175  * \ref graphlab::dc_dist_object::future_remote_request() , but has the
176  * additional capability that if a \ref graphlab::request_future::wait() is
177  * called on the request while within a fiber, it deschedules the fiber and
178  * context switches, returning only when the future is ready. This allows
179  * the future to be used from within a fiber.
180  *
181  * Since this function is not a member of the \ref dc_dist_object class,
182  * it needs to be provided a reference to the owning object's dc_dist_object.
183  *
184  * \ref graphlab::fiber_remote_request is the version of this function
185  * for remotely calling global functions.
186  *
187  * Example:
188  * \code
189  * // A print function is defined in the distributed object
190  * class distributed_obj_example {
191  * graphlab::dc_dist_object<distributed_obj_example> rmi;
192  * ... initialization and constructor ...
193  * private:
194  * int add_one(int i) {
195  * return i + 1;
196  * }
197  * public:
198  * int add_one_from_machine_1(int i) {
199  * // calls the add_one function on machine 1 with the argument i
200  * // this call returns immediately
201  * graphlab::request_future<int> future =
202  * object_future_remote_request(rmi, 1, &distributed_obj_example::add_one, i);
203  *
204  * // ... we can do other stuff here
205  * // then when we want the answer
206  * // this is safe to do within a fiber as it will not halt other fibers.
207  * int result = future();
208  * return result;
209  * }
210  * }
211  * \endcode
212  *
213  * \see graphlab::dc_dist_object::remote_request
214  * graphlab::dc_dist_object::future_remote_request
215  * graphlab::fiber_remote_request
216  *
217  * \param rmiobj The dc_dist_object to use to send the request.
218  * \param targetmachine The ID of the machine to run the function on
219  * \param fn The function to run on the target machine. Must be a pointer to
220  * member function in the owning object.
221  * \param ... The arguments to send to Fn. Arguments must be serializable.
222  * and must be castable to the target types.
223  *
224  * \returns Returns a future templated around the same type as the return
225  * value of the called function
226  */
227  request_future<RetVal> object_fiber_remote_request(dc_dist_object<T> rmiobj,
228  procid_t targetmachine,
229  Fn fn, ...);
