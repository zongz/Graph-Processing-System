41  /**
42  * A hopscotch hash set. More or less similar
43  * interface as boost::unordered_set, not necessarily
44  * entirely STL compliant.
45  * Really should only be used to store small keys and trivial values.
46  *
47  * \tparam Key The key of the set
48  * \tparam Hash The hash functor type. Defaults to std::hash<Key> if C++11 is
49  * available. Otherwise defaults to boost::hash<Key>
50  * \tparam KeyEqual The functor used to identify object equality. Defaults to
51  * std::equal_to<Key>
52  */
53  template <typename Key,
54  typename Hash = _HOPSCOTCH_SET_DEFAULT_HASH,
55  typename KeyEqual = std::equal_to<Key> >

101  // rehashes the hash table to one which is double the size
102  container_type* rehash_to_new_container(size_t newsize = (size_t)(-1)) {
103  /*
104  std::cerr << "Rehash at " << container->size() << "/"
105  << container->capacity() << ": "
106  << container->load_factor() << std::endl;
107  */
108  // rehash
109  if (newsize == (size_t)(-1)) newsize = container->size() * 2;
110  container_type* newcontainer = create_new_container(newsize);
111  const_iterator citer = begin();
112  while (citer != end()) {
113  assert(newcontainer->insert(*citer) != newcontainer->end());
114  ++citer;
115  }
116  return newcontainer;
117  }
118 
119  // Inserts a value into the hash table. This does not check
120  // if the key already exists, and may produce duplicate values.
121  iterator do_insert(const value_type &v) {
122  iterator iter = container->insert(v);
123 
124  if (iter != container->end()) {
125  return iter;
126  }
127  else {
128  container_type* newcontainer = rehash_to_new_container();
129  iter = newcontainer->insert(v);
130  assert(iter != newcontainer->end());
131  std::swap(container, newcontainer);
132  delete newcontainer;
133  return iter;
134  }
135  }
