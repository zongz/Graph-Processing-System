163  /** \brief Finalize completes the local graph data structure
164  * and the vertex record information.
165  *
166  * \internal
167  * The finalization goes through 5 steps:
168  *
169  * 1. Construct local graph using the received edges, during which
170  * the vid2lvid map is built.
171  *
172  * 2. Construct lvid2record map (of empty entries) using the received vertices.
173  *
174  * 3. Complete lvid2record map by exchanging the vertex_info.
175  *
176  * 4. Exchange the negotiation records, including singletons. (Local graph
177  * handling singletons).
178  *
179  * 5. Exchange global graph statistics.
180  */
181  virtual void finalize() {
182 
183  rpc.full_barrier();
184 
185  bool first_time_finalize = false;
186  /**
187  * Fast pass for first time finalization.
188  */
189  if (graph.is_dynamic()) {
190  size_t nverts = graph.num_local_vertices();
191  rpc.all_reduce(nverts);
192  first_time_finalize = (nverts == 0);
193  } else {
194  first_time_finalize = false;
195  }
196 
197 
198  if (rpc.procid() == 0) {
199  logstream(LOG_EMPH) << "Finalizing Graph..." << std::endl;
200  }

209  vertex_buffer_type;

211  /**
212  * \internal
213  * Buffer storage for new vertices to the local graph.
214  */
215  typedef typename graph_type::hopscotch_map_type vid2lvid_map_type;
216  vid2lvid_map_type vid2lvid_buffer;

218  /**
219  * \internal
220  * The begining id assinged to the first new vertex.
221  */
222  const lvid_type lvid_start = graph.vid2lvid.size();
223 
224  /**
225  * \internal
226  * Bit field incidate the vertex that is updated during the ingress.
227  */
228  dense_bitset updated_lvids(graph.vid2lvid.size());

230  /**************************************************************************/
231  /* */
232  /* Flush any additional data */
233  /* */
234  /**************************************************************************/
235  edge_exchange.flush(); vertex_exchange.flush();
236 
237  /**
238  * Fast pass for redundant finalization with no graph changes.
239  */
240  {
241  size_t changed_size = edge_exchange.size() + vertex_exchange.size();
242  rpc.all_reduce(changed_size);
243  if (changed_size == 0) {
244  logstream(LOG_INFO) << "Skipping Graph Finalization because no changes happened..." << std::endl;
245  return;
246  }
247  }
248 
249  if(rpc.procid() == 0)
250  memory_info::log_usage("Post Flush");
251 
252 
253  /**************************************************************************/
254  /* */
255  /* Construct local graph */
256  /* */
257  /**************************************************************************/
258  { // Add all the edges to the local graph
259  logstream(LOG_INFO) << "Graph Finalize: constructing local graph" << std::endl;
260  const size_t nedges = edge_exchange.size()+1;
261  graph.local_graph.reserve_edge_space(nedges + 1);
262  edge_buffer_type edge_buffer;
263  procid_t proc;
264  while(edge_exchange.recv(proc, edge_buffer)) {
265  foreach(const edge_buffer_record& rec, edge_buffer) {
266  // Get the source_vlid;
267  lvid_type source_lvid(-1);
268  if(graph.vid2lvid.find(rec.source) == graph.vid2lvid.end()) {
269  if (vid2lvid_buffer.find(rec.source) == vid2lvid_buffer.end()) {
270  source_lvid = lvid_start + vid2lvid_buffer.size();
271  vid2lvid_buffer[rec.source] = source_lvid;
272  } else {
273  source_lvid = vid2lvid_buffer[rec.source];
274  }
275  } else {
276  source_lvid = graph.vid2lvid[rec.source];
277  updated_lvids.set_bit(source_lvid);
278  }
279  // Get the target_lvid;
280  lvid_type target_lvid(-1);
281  if(graph.vid2lvid.find(rec.target) == graph.vid2lvid.end()) {
282  if (vid2lvid_buffer.find(rec.target) == vid2lvid_buffer.end()) {
283  target_lvid = lvid_start + vid2lvid_buffer.size();
284  vid2lvid_buffer[rec.target] = target_lvid;
285  } else {
286  target_lvid = vid2lvid_buffer[rec.target];
287  }
288  } else {
289  target_lvid = graph.vid2lvid[rec.target];
290  updated_lvids.set_bit(target_lvid);
291  }
292  graph.local_graph.add_edge(source_lvid, target_lvid, rec.edata);
293  // std::cout << "add edge " << rec.source << "\t" << rec.target << std::endl;
294  } // end of loop over add edges
295  } // end for loop over buffers
296  edge_exchange.clear();
297 
298  ASSERT_EQ(graph.vid2lvid.size() + vid2lvid_buffer.size(), graph.local_graph.num_vertices());
299  if(rpc.procid() == 0) {
300  memory_info::log_usage("Finished populating local graph.");
301  }
302 
303  // Finalize local graph
304  logstream(LOG_INFO) << "Graph Finalize: finalizing local graph."
305  << std::endl;
306  graph.local_graph.finalize();
307  logstream(LOG_INFO) << "Local graph info: " << std::endl
308  << "\t nverts: " << graph.local_graph.num_vertices()
309  << std::endl
310  << "\t nedges: " << graph.local_graph.num_edges()
311  << std::endl;
312 
313  if(rpc.procid() == 0) {
314  memory_info::log_usage("Finished finalizing local graph.");
315  // debug
316  // std::cout << graph.local_graph << std::endl;
317  }
318  }

320  /**************************************************************************/
321  /* */
322  /* Receive and add vertex data to masters */
323  /* */
324  /**************************************************************************/
325  // Setup the map containing all the vertices being negotiated by this machine
326  { // Receive any vertex data sent by other machines
327  vertex_buffer_type vertex_buffer; procid_t sending_proc(-1);
328  while(vertex_exchange.recv(sending_proc, vertex_buffer)) {
329  foreach(const vertex_buffer_record& rec, vertex_buffer) {
330  lvid_type lvid(-1);
331  if (graph.vid2lvid.find(rec.vid) == graph.vid2lvid.end()) {
332  if (vid2lvid_buffer.find(rec.vid) == vid2lvid_buffer.end()) {
333  lvid = lvid_start + vid2lvid_buffer.size();
334  vid2lvid_buffer[rec.vid] = lvid;
335  } else {
336  lvid = vid2lvid_buffer[rec.vid];
337  }
338  } else {
339  lvid = graph.vid2lvid[rec.vid];
340  updated_lvids.set_bit(lvid);
341  }
342  if (vertex_combine_strategy && lvid < graph.num_local_vertices()) {
343  vertex_combine_strategy(graph.l_vertex(lvid).data(), rec.vdata);
344  } else {
345  graph.local_graph.add_vertex(lvid, rec.vdata);
346  }
347  }
348  }
349  vertex_exchange.clear();
350  if(rpc.procid() == 0)
351  memory_info::log_usage("Finished adding vertex data");
352  } // end of loop to populate vrecmap

356  /**************************************************************************/
357  /* */
358  /* assign vertex data and allocate vertex (meta)data space */
359  /* */
360  /**************************************************************************/
361  { // Determine masters for all negotiated vertices
362  const size_t local_nverts = graph.vid2lvid.size() + vid2lvid_buffer.size();
363  graph.lvid2record.reserve(local_nverts);
364  graph.lvid2record.resize(local_nverts);
365  graph.local_graph.resize(local_nverts);
366  foreach(const vid2lvid_pair_type& pair, vid2lvid_buffer) {
367  vertex_record& vrec = graph.lvid2record[pair.second];
368  vrec.gvid = pair.first;
369  vrec.owner = graph_hash::hash_vertex(pair.first) % rpc.numprocs();
370  }
371  ASSERT_EQ(local_nverts, graph.local_graph.num_vertices());
372  ASSERT_EQ(graph.lvid2record.size(), graph.local_graph.num_vertices());
373  if(rpc.procid() == 0)
374  memory_info::log_usage("Finihsed allocating lvid2record");
375  }

377  /**************************************************************************/
378  /* */
379  /* Master handshake */
380  /* */
381  /**************************************************************************/
382  {
383 #ifdef _OPENMP
384  buffered_exchange<vertex_id_type> vid_buffer(rpc.dc(), omp_get_max_threads());
385 #else
386  buffered_exchange<vertex_id_type> vid_buffer(rpc.dc());
387 #endif
388 
389 #ifdef _OPENMP
390 #pragma omp parallel for
391 #endif
392  // send not owned vids to their master
393  for (lvid_type i = lvid_start; i < graph.lvid2record.size(); ++i) {
394  procid_t master = graph.lvid2record[i].owner;
395  if (master != rpc.procid())
396 #ifdef _OPENMP
397  vid_buffer.send(master, graph.lvid2record[i].gvid, omp_get_thread_num());
398 #else
399  vid_buffer.send(master, graph.lvid2record[i].gvid);
400 #endif
401  }
402  vid_buffer.flush();
403  rpc.barrier();
404 
405  // receive all vids owned by me
406  mutex flying_vids_lock;
407  boost::unordered_map<vertex_id_type, mirror_type> flying_vids;
408 #ifdef _OPENMP
409 #pragma omp parallel
410 #endif
411  {
412  typename buffered_exchange<vertex_id_type>::buffer_type buffer;
413  procid_t recvid;
414  while(vid_buffer.recv(recvid, buffer)) {
415  foreach(const vertex_id_type vid, buffer) {
416  if (graph.vid2lvid.find(vid) == graph.vid2lvid.end()) {
417  if (vid2lvid_buffer.find(vid) == vid2lvid_buffer.end()) {
418  flying_vids_lock.lock();
419  mirror_type& mirrors = flying_vids[vid];
420  flying_vids_lock.unlock();
421  mirrors.set_bit(recvid);
422  } else {
423  lvid_type lvid = vid2lvid_buffer[vid];
424  graph.lvid2record[lvid]._mirrors.set_bit(recvid);
425  }
426  } else {
427  lvid_type lvid = graph.vid2lvid[vid];
428  graph.lvid2record[lvid]._mirrors.set_bit(recvid);
429  updated_lvids.set_bit(lvid);
430  }
431  }
432  }
433  }
434 
435  vid_buffer.clear();
436  // reallocate spaces for the flying vertices.
437  size_t vsize_old = graph.lvid2record.size();
438  size_t vsize_new = vsize_old + flying_vids.size();
439  graph.lvid2record.resize(vsize_new);
440  graph.local_graph.resize(vsize_new);
441  for (typename boost::unordered_map<vertex_id_type, mirror_type>::iterator it = flying_vids.begin();
442  it != flying_vids.end(); ++it) {
443  lvid_type lvid = lvid_start + vid2lvid_buffer.size();
444  vertex_id_type gvid = it->first;
445  graph.lvid2record[lvid].owner = rpc.procid();
446  graph.lvid2record[lvid].gvid = gvid;
447  graph.lvid2record[lvid]._mirrors= it->second;
448  vid2lvid_buffer[gvid] = lvid;
449  // std::cout << "proc " << rpc.procid() << " recevies flying vertex " << gvid << std::endl;
450  }
451  } // end of master handshake
452 
453  /**************************************************************************/
454  /* */
455  /* Merge in vid2lvid_buffer */
456  /* */
457  /**************************************************************************/
458  {
459  if (graph.vid2lvid.size() == 0) {
460  graph.vid2lvid.swap(vid2lvid_buffer);
461  } else {
462  graph.vid2lvid.rehash(graph.vid2lvid.size() + vid2lvid_buffer.size());
463  foreach (const typename vid2lvid_map_type::value_type& pair, vid2lvid_buffer) {
464  graph.vid2lvid.insert(pair);
465  }
466  vid2lvid_buffer.clear();
467  // vid2lvid_buffer.swap(vid2lvid_map_type(-1));
468  }
469  }

473  /* */
474  /* synchronize vertex data and meta information */
475  /* */
476  /**************************************************************************/
477  {
478  // construct the vertex set of changed vertices
479 
480  // Fast pass for first time finalize;
481  vertex_set changed_vset(true);
482 
483  // Compute the vertices that needs synchronization
484  if (!first_time_finalize) {
485  vertex_set changed_vset = vertex_set(false);
486  changed_vset.make_explicit(graph);
487  updated_lvids.resize(graph.num_local_vertices());

490  }

512  /* Exchange graph statistics among all nodes and compute
513  * global statistics for the distributed graph. */
514  void exchange_global_info () {
515  // Count the number of vertices owned locally
516  graph.local_own_nverts = 0;
517  foreach(const vertex_record& record, graph.lvid2record)
518  if(record.owner == rpc.procid()) ++graph.local_own_nverts;
519 
520  // Finalize global graph statistics.
521  logstream(LOG_INFO)
522  << "Graph Finalize: exchange global statistics " << std::endl;
523 
524  // Compute edge counts
525  std::vector<size_t> swap_counts(rpc.numprocs());
526  swap_counts[rpc.procid()] = graph.num_local_edges();
527  rpc.all_gather(swap_counts);
528  graph.nedges = 0;
529  foreach(size_t count, swap_counts) graph.nedges += count;
530 
531 
532  // compute vertex count
533  swap_counts[rpc.procid()] = graph.num_local_own_vertices();
534  rpc.all_gather(swap_counts);
535  graph.nverts = 0;
536  foreach(size_t count, swap_counts) graph.nverts += count;
537 
538  // compute replicas
539  swap_counts[rpc.procid()] = graph.num_local_vertices();
540  rpc.all_gather(swap_counts);
541  graph.nreplicas = 0;
542  foreach(size_t count, swap_counts) graph.nreplicas += count;
543 
544 
545  if (rpc.procid() == 0) {
546  logstream(LOG_EMPH) << "Graph info: "
547  << "\n\t nverts: " << graph.num_vertices()
548  << "\n\t nedges: " << graph.num_edges()
549  << "\n\t nreplicas: " << graph.nreplicas
550  << "\n\t replication factor: " << (double)graph.nreplicas/graph.num_vertices()
551  << std::endl;
552  }
