297  /**
298  * Atomically deschedules the current thread and unlocks the mutex.
299  *
300  * deschedule_self() and schedule_tid() must be managed carefully
301  * to avoid race conditions. i.e. schedule_tid() happending before
302  * deschedule_self().
303  *
304  * To support this correctly, the descheduling must be paired together
305  * with a mutex.
306  *
307  * For instance, to use this to implement a promise.
308  * \code
309  * // descheduling fiber
310  * pthread_mutex_lock(&lock);
311  * if ( ... promise not ready ...) {
312  * deschedule_self(&lock);
313  * } else {
314  * pthread_mutex_unlock(&lock);
315  * }
316  * ... use the promise ...
317  * \endcode
318  *
319  *
320  * The promise execution thread then must do the following
321  * \code
322  * ... tid contains the fiber ID to wake when promise is done
323  * ... compute promise...
324  * pthread_mutex_lock(&lock); // same lock as above
325  * ... set promise completion...
326  * schedule_tid(tid); // wake up the fiber
327  * pthread_mutex_unlock(&lock);
328  * \endcode
329  */
330  static void deschedule_self(pthread_mutex_t* lock);
331 
332  /**
333  * Schedules a fiber for execution.
334  * If this fiber was previously descheduled by
335  * deschedule_self(), the fiber is scheduled for execution.
336  * Otherwise, nothing happens. Some care must be taken to avoid race
337  * conditions. See the deschedule_self() function for details.
338  * This thread by default will be stuck at the head of queue
339  * and will wake up quickly.
340  *
341  * \param priority If true, thread will be placed at the head
342  * of the scheduler. If false, it will be placed at the tail
343  * of the scheduler
344  */
345  static void schedule_tid(size_t tid, bool priority = true);
