37  /**
38  * \ingroup rpc
39  *
40  * The buffered exchange provides high performance exchange of bulk data
41  * between machines. This is like the \ref graphlab::buffered_exchange, but is
42  * much stricter. All send calls must occur within fibers, and all recv calls
43  * must occur within fibers. Specifically, a collection of "fiber-worker-local"
44  * send buffer and receive buffers are created. Sends by fibers write into the
45  * buffer owned by the current worker handling the fiber. Receives similarly
46  * read from the buffer owned by the current worker handling the fiber.
47  * As such a bit more subtlety is needed to use this class correctly.
48  *
49  * For instance, if we are doing bulk exchanges of integers:
50  * \code
51  * buffered_exchange<int> exchange(dc);
52  * .. In parallel in fibers .. {
53  * exchange.send([target machine], [value to send to target])
54  * exchange.partial_flush();
55  * }
56  *
57  * .. now in 1 thread ..
58  * exchange.flush()
59  *
60  * .. In parallel in fibers .. {
61  * procid_t proc;
62  * std::vector<buffered_exchange<int>::buffer_record> buffer; // (an array of buffers)
63  * while(exchange.recv(buffer)) {
64  * for each buffer_record in buffer:
65  * buffer_record.proc is the machine which sent the contents of this record
66  * buffer_record.buffer is an array containing values sent by the machine buffer_record.proc
67  * }
68  * }
69  *
70  * .. now in 1 thread ..
71  * exchange.recv(buffer, false); // get from all receive buffers
72  * while(exchange.recv(buffer)) {
73  * for each buffer_record in buffer:
74  * buffer_record.proc is the machine which sent the contents of this record
75  * buffer_record.buffer is an array containing values sent by the machine buffer_record.proc
76  * }
77  * \endcode
78  *
79  * \note The buffered exchange sends data in the background, so recv can be
80  * called even before the flush calls.
81  * \note The last single threaded receive is not necessary if worker-affinity
82  * is set correctly so that every worker is active in the parallel receiving
83  * block.
84  *
85  * \see graphlab::buffered_exchange
86  */

118  /**
119  * Flushes the send buffer local to worker id "wid" and going to process proc
120  */
121  void flush_buffer(size_t wid, procid_t proc) {
122  if(send_buffers[wid][proc].oarc) {
123  // write the length at the end of the buffere are returning
124  send_buffers[wid][proc].oarc->write(reinterpret_cast<char*>(&send_buffers[wid][proc].numinserts), sizeof(size_t));
125  rpc.split_call_end(proc, send_buffers[wid][proc].oarc);
126 // logstream(LOG_DEBUG) << rpc.procid() << ": Sending exchange of length "
127 // << send_buffers[wid][proc].oarc->off << " to "
128 // << proc << std::endl;
129  send_buffers[wid][proc].oarc = NULL;
130  send_buffers[wid][proc].numinserts = 0;
131  }
132  }

135  /**
136  * Constructs a buffered exchange object.
137  *
138  * \ref dc The master distributed_control object
139  * \ref max_buffer_size The size of the per thread and per target send buffer.
140  */
141  fiber_buffered_exchange(distributed_control& dc,
142  const size_t max_buffer_size = DEFAULT_BUFFERED_EXCHANGE_SIZE) :
143  rpc(dc, this),
144  max_buffer_size(max_buffer_size) {
145  send_buffers.resize(fiber_control::get_instance().num_workers());
146  recv_buffers.resize(fiber_control::get_instance().num_workers());
147  for (size_t i = 0;i < send_buffers.size(); ++i) {
148  send_buffers[i].resize(dc.numprocs());
149  for (size_t j = 0;j < send_buffers[i].size(); ++j) {

172  /**
173  * Sends a value to a target machine.
174  * Must be called from within a fiber
175  */
176  void send(const procid_t proc, const T& value) {
177  size_t wid = fiber_control::get_worker_id();
178  if (send_buffers[wid][proc].oarc == NULL) {
179  send_buffers[wid][proc].oarc = rpc.split_call_begin(&fiber_buffered_exchange::rpc_recv);
180  // write a header
181  (*send_buffers[wid][proc].oarc) << rpc.procid();
182  send_buffers[wid][proc].numinserts = 0;
183  }
184 
185  (*(send_buffers[wid][proc].oarc)) << value;
186  ++send_buffers[wid][proc].numinserts;
187 
188 
189  if(send_buffers[wid][proc].oarc->off >= max_buffer_size) {
190  flush_buffer(wid, proc);
191  }
192  } // end of send

194  /**
195  * Flushes the send buffers owned by the worker currently running the
196  * current fiber.

198  void partial_flush() {

204  /**
205  * Flushes all send buffers. Must be called only on one thread.
206  * Will not return until all machines call flush.
207  */
208  void flush() {
209  for(size_t i = 0; i < send_buffers.size(); ++i) {
210  for (size_t j = 0;j < send_buffers[i].size(); ++j) {
211  flush_buffer(i,j);
212  }
213  }
214  rpc.dc().flush();
215  rpc.full_barrier();
216  } // end of flush

218  /**
219  * Receives a collection of buffers.
220  * Must be called from within a fiber.
221  * \param ret_buffer If return value is true, this contains a collection of
222  * buffers sent to this machine.
223  * \param self_buffer If true, only receives the worker local buffers.
224  * \returns true If ret_buffer contains values.
225  */
226  bool recv(std::vector<buffer_record>& ret_buffer,
227  const bool self_buffer = true) {
228  fiber_control::fast_yield();
229  ret_buffer.clear();
230  bool success = false;
231  if (self_buffer) {
232  // get from my own buffer
233  size_t wid = fiber_control::get_worker_id();
234  lock.lock();
235  if(!recv_buffers[wid].empty()) {
236  success = true;
237  std::swap(ret_buffer, recv_buffers[wid]);
238  } else {
239  for (size_t i = 0;i < recv_buffers.size(); ++i) {
240  if(!recv_buffers[i].empty()) {
241  success = true;
242  std::swap(ret_buffer, recv_buffers[i]);
243  break;
244  }
245  }
246  }
247  lock.unlock();
248  } else {
249  for (size_t i = 0;i < recv_buffers.size(); ++i) {
250  if(!recv_buffers[i].empty()) {
251  success = true;
252  std::swap(ret_buffer, recv_buffers[i]);
253  break;
254  }
255  }
256  }
257  return success;
258  } // end of recv

287  void rpc_recv(size_t len, wild_pointer w) {
288  buffer_type tmp;
289  iarchive iarc(reinterpret_cast<const char*>(w.ptr), len);
290  // first desrialize the source process
291  procid_t src_proc; iarc >> src_proc;
292 // logstream(LOG_DEBUG) << rpc.procid() << ": Receiving exchange of length "
293 // << len << " from " << src_proc << std::endl;
294  // create an iarchive which just points to the last size_t bytes
295  // to get the number of elements
296  iarchive numel_iarc(reinterpret_cast<const char*>(w.ptr) + len - sizeof(size_t),
297  sizeof(size_t));
298  size_t numel = 0;
299  numel_iarc.read(reinterpret_cast<char*>(&numel), sizeof(size_t));
300  //std::cout << "Receiving: " << numel << "\n";
301  tmp.resize(numel);
302  for (size_t i = 0;i < numel; ++i) {
303  iarc >> tmp[i];
304  }
305 
306  size_t wid = fiber_control::get_worker_id();
307  lock.lock();
308  recv_buffers[wid].push_back(buffer_record());
309  buffer_record& rec = recv_buffers[wid].back();
310  rec.proc = src_proc;
311  rec.buffer.swap(tmp);
312  lock.unlock();
313  } // end of rpc rcv
