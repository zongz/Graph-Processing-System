54 void pagerank(warp_engine_type::context& context,
55  graph_type::vertex_type vertex) {
56  float old_vdata = vertex.data();
57  vertex.data() = 0.15 + 0.85 * warp::map_reduce_neighborhood(vertex,
58  IN_EDGES,
59  pagerank_map);
60  if (std::fabs(old_vdata - vertex.data()) > 1E-2) {
61  warp::broadcast_neighborhood(context, vertex, OUT_EDGES, signal_neighbor);
62  }
63 }

84 int main(int argc, char** argv) {
85  global_logger().set_log_level(LOG_INFO);
86  // Initialize control plain using mpi
87  mpi_tools::init(argc, argv);
88  distributed_control dc;
89 
90  // Parse command line options -----------------------------------------------
91  command_line_options clopts("PageRank algorithm.");
92  std::string graph_dir;
93  std::string format = "tsv";
94  clopts.attach_option("graph", graph_dir, "The graph file. ");
95  clopts.attach_option("format", format, "The graph file format");
96  size_t iterations = 10;
97  clopts.attach_option("iterations", iterations,
98  "Number of asynchronous iterations to run");
99  std::string saveprefix;
100  clopts.attach_option("saveprefix", saveprefix,
101  "Prefix to save the output pagerank in");
102 
103  if(!clopts.parse(argc, argv)) {
104  dc.cout() << "Error in parsing command line arguments." << std::endl;
105  return EXIT_FAILURE;
106  }
107 
108  // Build the graph ----------------------------------------------------------
109  graph_type graph(dc, clopts);
110  dc.cout() << "Loading graph in format: "<< format << std::endl;
111  graph.load_format(graph_dir, format);
112  // must call finalize before querying the graph
113  graph.finalize();
114 
115  // Initialize the vertex data
116  graph.transform_vertices(init_vertex);
117 
118  timer ti;
119  warp_engine_type engine(dc, graph, clopts);
120  engine.signal_all();
121  engine.set_update_function(pagerank);
122  engine.start();
123 
124  dc.cout() << "Finished Running in " << ti.current_time()
125  << " seconds." << std::endl;
126 
127 
128  // Save the final graph -----------------------------------------------------
129  if (saveprefix != "") {
130  graph.save(saveprefix, pagerank_writer(),
131  false, // do not gzip
132  true, // save vertices
133  false); // do not save edges
134  }
135 
136  mpi_tools::finalize();
137 } // End of main
