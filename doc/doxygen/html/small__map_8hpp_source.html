136  //! Lookup an element in the map
137  inline mapped_type& operator[](const key_type& key) {
138  value_type* ptr =

140  std::make_pair(key, mapped_type()),
141  less());
142  if(ptr != end() && (key == ptr->first) ) { return ptr->second; }
