61  static void basic_local_transform_neighborhood(GraphType& graph,
62  edge_dir_type edge_direction,
63  void(*transform_fn)(edge_type edge,
64  vertex_type other),
65  vertex_id_type vid) {
66  lvid_type lvid = graph.local_vid(vid);
67  local_vertex_type local_vertex(graph.l_vertex(lvid));
68 
69  if(edge_direction == IN_EDGES || edge_direction == ALL_EDGES) {
70  foreach(local_edge_type local_edge, local_vertex.in_edges()) {
71  edge_type edge(local_edge);
72  vertex_type other(local_edge.source());
73  lvid_type a = edge.source().local_id(), b = edge.target().local_id();
74  graph.get_lock_manager()[std::min(a,b)].lock();
75  graph.get_lock_manager()[std::max(a,b)].lock();
76  transform_fn(edge, other);
77  graph.get_lock_manager()[a].unlock();
78  graph.get_lock_manager()[b].unlock();
79  }
80  }
81  // do out edges
82  if(edge_direction == OUT_EDGES || edge_direction == ALL_EDGES) {
83  foreach(local_edge_type local_edge, local_vertex.out_edges()) {
84  edge_type edge(local_edge);
85  vertex_type other(local_edge.target());
86  lvid_type a = edge.source().local_id(), b = edge.target().local_id();
87  graph.get_lock_manager()[std::min(a,b)].lock();
88  graph.get_lock_manager()[std::max(a,b)].lock();
89  transform_fn(edge, other);
90  graph.get_lock_manager()[a].unlock();
91  graph.get_lock_manager()[b].unlock();
92  }
93  }
94  }

111  static void basic_transform_neighborhood(typename GraphType::vertex_type current,
112  edge_dir_type edge_direction,
113  void(*transform_fn)(edge_type edge, vertex_type other)) {
114  // get a reference to the graph
115  GraphType& graph = current.graph_ref;
116  // get the object ID of the graph
117  size_t objid = graph.get_rpc_obj_id();
118  typename GraphType::vertex_record vrecord = graph.l_get_vertex_record(current.local_id());
119 
120  // make sure we are running on a master vertex
121  ASSERT_EQ(vrecord.owner, distributed_control::get_instance_procid());
122 
123  // create num-mirrors worth of requests
124  std::vector<request_future<void > > requests(vrecord.num_mirrors());
125 
126  size_t ctr = 0;
127  foreach(procid_t proc, vrecord.mirrors()) {
128  // issue the communication
129  requests[ctr] = fiber_remote_request(proc,
130  transform_neighborhood_impl<GraphType>::basic_local_transform_neighborhood_from_remote,
131  objid,
132  edge_direction,
133  reinterpret_cast<size_t>(transform_fn),
134  current.id());
135  ++ctr;
136  }
137  // compute the local tasks
138  basic_local_transform_neighborhood(graph,
139  edge_direction,
140  transform_fn,
141  current.id());
142  // now, wait for everyone
143  for (size_t i = 0;i < requests.size(); ++i) {
144  requests[i]();
145  }
146  }

271 /**
272  * \ingroup warp
273  *
274  * The transform_neighborhood() function allows a parallel transformation of the
275  * neighborhood of a vertex to be performed. This is a blocking operation, and
276  * will not return until the distributed computation is complete. When run
277  * inside a fiber, to hide latency, the system will automatically context
278  * switch to evaluate some other fiber which is ready to run.
279  *
280  * Abstractly, the computation accomplishes the following:
281  *
282  * \code
283  * foreach(edge in neighborhood of current vertex) {
284  * transform_fn(edge, opposite vertex)
285  * }
286  * \endcode
287  *
288  * \attention It is important that the transform_fn should only make modifications to the
289  * edge data, and not the data on the other vertex.
290  *
291  * \attention This call does not accomplish synchronization, thus
292  * modifications to the current vertex will not be reflected during
293  * the call. In other words, inside the mapper function, only the values on
294  * edge.data() and other.data() will be valid. The value of the vertex
295  * on the "self" end of the edge will not reflect changes you made to the vertex
296  * immediately before calling transform_neighborhood(). Use the overload of
297  * transform_neighborhood (below) if you want to pass on additional information to the
298  * mapper.
299  *
300  * Here is an example which clears all the in edge values of some set of vertices.
301  * \code
302  * void clear_value(graph_type::edge_type edge, graph_type::vertex_type other) {
303  * edge.data() = 0;
304  * }
305  *
306  *
307  * void clear_in_edges(graph_type::vertex_type vertex) {
308  * warp::transform_neighborhood(vertex,
309  * IN_EDGES,
310  * clear_value);
311  * }
312  *
313  * ...
314  * warp::parfor_all_vertices(graph, clear_in_edges, some_vset);
315  * \endcode
316  *
317  *
318  * An overload is provided which allows you to pass an additional arbitrary
319  * argument to the transform
320  *
321  * \param current The vertex to map reduce the neighborhood over
322  * \param edge_direction To run over all IN_EDGES, OUT_EDGES or ALL_EDGES
323  * \param transform_fn The transform function to run on all the selected edges.
324  *
325  * \see warp_engine
326  * \see warp::parfor_all_vertices()
327  * \see warp::map_reduce_neighborhood()
328  * \see warp::broadcast_neighborhood()
329  */
330 template <typename VertexType>
331 void transform_neighborhood(VertexType current,
332  edge_dir_type edge_direction,
333  void (*transform_fn)(typename VertexType::graph_type::edge_type edge,
334  VertexType other)) {
335  warp_impl::
336  transform_neighborhood_impl<typename VertexType::graph_type>::
337  basic_transform_neighborhood(current, edge_direction,
338  transform_fn);
339 }

342 /**
343  * \ingroup warp
344  *
345  * The transform_neighborhood function allows a parallel transformation of the
346  * neighborhood of a vertex to be performed. This is a blocking operation, and
347  * will not return until the distributed computation is complete. When run
348  * inside a fiber, to hide latency, the system will automatically context
349  * switch to evaluate some other fiber which is ready to run.
350  *
351  * This is the more general overload of the
352  * warp::transform_neighborhood() function
353  * which allows an additional arbitrary extra argument to be passed along
354  * to the mapper and combiner functions.
355  *
356  * Abstractly, the computation accomplishes the following:
357  *
358  * \code
359  * foreach(edge in neighborhood of current vertex) {
360  * transform_fn(edge, extraarg, opposite vertex)
361  * }
362  * \endcode
363  *
364  * \attention It is important that the transform_fn should only make modifications to the
365  * edge data, and not the data on the other vertex.
366  *
367  * \attention This call does not accomplish synchronization, thus
368  * modifications to the current vertex will not be reflected during
369  * the call. In other words, inside the mapper function, only the values on
370  * edge.data() and other.data() will be valid. The value of the vertex
371  * on the "self" end of the edge will not reflect changes you made to the vertex
372  * immediately before calling warp::transform_neighborhood() . Use the overload of
373  * warp::transform_neighborhood() if you want to pass on additional information to the
374  * mapper.
375  *
376  * Here is an example which set all the in edge values of some set of vertices
377  * to the source vertex's value.
378  * \code
379  * void set_value(graph_type::edge_type edge, graph_type::vertex_type other,
380  * int value) {
381  * edge.data() = value;
382  * }
383  *
384  *
385  * void set_in_edges(graph_type::vertex_type vertex) {
386  * warp::transform_neighborhood(vertex,
387  * IN_EDGES,
388  * clear_value,
389  * vertex.data(), // say this is an integer
390  * );
391  * }
392  *
393  * ...
394  * warp::parfor_all_vertices(graph, set_in_edges, some_vset);
395  * \endcode
396  *
397  *
398  * \param current The vertex to map reduce the neighborhood over
399  * \param edge_direction To run over all IN_EDGES, OUT_EDGES or ALL_EDGES
400  * \param extra An additional argument to be passed to the mapper and combiner functions.
401  * \param transform_fn The transform function to run on all the selected edges.
402  *
403  * \see warp_engine
404  * \see warp::parfor_all_vertices()
405  * \see warp::map_reduce_neighborhood()
406  * \see warp::broadcast_neighborhood()
407  */
408 template <typename ExtraArg, typename VertexType>
409 void transform_neighborhood(VertexType current,
410  edge_dir_type edge_direction,
411  void(*transform_fn)(typename VertexType::graph_type::edge_type edge,
412  VertexType other,
413  const ExtraArg extra),
414  const ExtraArg extra) {
415  warp_impl::
416  transform_neighborhood_impl2<typename VertexType::graph_type, ExtraArg>::
417  extended_transform_neighborhood(current, edge_direction,
418  transform_fn,
419  extra);
420 }
