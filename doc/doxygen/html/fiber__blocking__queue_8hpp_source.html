40  * \ingroup util
41  * \brief Implements a blocking queue useful for producer/consumer models.
42  * Similar to the \ref blocking_queue, but requires all threads waiting on
43  * the queue to be fibers. Also only supports the basic wait on dequeue
44  * operation, and not the other timed_wait, or wait_until_empty operations
45  * supported by the \ref blocking_queue
46  */

48  class fiber_blocking_queue {

64  fiber_blocking_queue() : m_alive(true),sleeping(0) { }

66  void wake_a_fiber() {
67  if (!fiber_queue.empty()) {
68  size_t fiber_id = fiber_queue.front();
69  fiber_queue.pop();
70  fiber_control::schedule_tid(fiber_id);
71  }
72  }

74  void wake_all_fibers() {
75  while(!fiber_queue.empty()) {
76  size_t fiber_id = fiber_queue.front();
77  fiber_queue.pop();
78  fiber_control::schedule_tid(fiber_id);
79  }
80  }

82  void fiber_sleep() {
83  fiber_queue.push(fiber_control::get_tid());
84  fiber_control::deschedule_self(&m_mutex.m_mut);
85  m_mutex.lock();

88  //! Add an element to the blocking queue
89  inline void enqueue(const T& elem, bool wake_consumer = true) {
90  m_mutex.lock();
91  m_queue.push_back(elem);
92  // Signal threads waiting on the queue
93  if (wake_consumer && sleeping) wake_a_fiber();
94  m_mutex.unlock();
95  }

122  inline bool wait_for_data() {
123  m_mutex.lock();
124  bool success = false;
125  // Wait while the queue is empty and this queue is alive
126  while(m_queue.empty() && m_alive) {
127  sleeping++;
128  fiber_sleep();
129  sleeping--;
130  }
131  // An element has been added or a signal was raised
132  if(!m_queue.empty()) {
133  success = true;
134  }
135  m_mutex.unlock();
136 
137  return success;
138  }

141  /**
142  * Blocks until an element is available in the queue
143  * or until stop_blocking() is called.
144  * The return value is a pair of <T value, bool success>
145  * If "success" if set, then "value" is valid and
146  * is an element popped from the queue.
147  * If "success" is false, stop_blocking() was called
148  * and the queue has been destroyed.
149  */
150  inline std::pair<T, bool> dequeue() {
151 
152  m_mutex.lock();
153  T elem = T();
154  bool success = false;
155  // Wait while the queue is empty and this queue is alive
156  while(m_queue.empty() && m_alive) {
157  sleeping++;
158  fiber_sleep();
159  sleeping--;
160  }
161  // An element has been added or a signal was raised
162  if(!m_queue.empty()) {
163  success = true;
164  elem = m_queue.front();
165  m_queue.pop_front();
166  }
167  m_mutex.unlock();
168 
169  return std::make_pair(elem, success);
170  }
171 
172  /**
173  * Returns an element if the queue has an entry.
174  * returns [item, false] otherwise.
175  */
176  inline std::pair<T, bool> try_dequeue() {
177  if (m_queue.empty() || m_alive == false) return std::make_pair(T(), false);
178  m_mutex.lock();
179  T elem = T();
180  // Wait while the queue is empty and this queue is alive
181  if (m_queue.empty() || m_alive == false) {
182  m_mutex.unlock();
183  return std::make_pair(elem, false);
184  }
185  else {
186  elem = m_queue.front();
187  m_queue.pop_front();
188  }
189  m_mutex.unlock();
190 
191  return std::make_pair(elem, true);
192  }
193 
194  //! Returns true if the queue is empty
195  inline bool empty() {
196  m_mutex.lock();
197  bool res = m_queue.empty();
198  m_mutex.unlock();
199  return res;
200  }
201 
202  /** Wakes up all threads waiting on the queue whether
203  or not an element is available. Once this function is called,
204  all existing and future dequeue operations will return with failure.
205  Note that there could be elements remaining in the queue after
206  stop_blocking() is called.
207  */
208  inline void stop_blocking() {
209  m_mutex.lock();
210  m_alive = false;
211  wake_all_fibers();
212  m_mutex.unlock();
213  }
214 
215  /**
216  Resumes operation of the fiber_blocking_queue. Future calls to
217  dequeue will proceed as normal.
218  */
219  inline void start_blocking() {
220  m_mutex.lock();
221  m_alive = true;
222  m_mutex.unlock();
223  }
224 
225  //! get the current size of the queue
226  inline size_t size() {
227  return m_queue.size();
228  }

231  /**
232  * Causes any threads currently blocking on a dequeue to wake up
233  * and evaluate the state of the queue. If the queue is empty,
234  * the threads will return back to sleep immediately. If the queue
235  * is destroyed through stop_blocking, all threads will return.
236  */
237  void broadcast() {
238  m_mutex.lock();
239  wake_all_fibers();
240  m_mutex.unlock();
241  }

244  ~fiber_blocking_queue() {
245  m_alive = false;

247  }
