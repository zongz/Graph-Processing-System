62  /**
63  * \ingroup engines
64  *
65  * \brief The synchronous engine executes all active vertex program
66  * synchronously in a sequence of super-step (iterations) in both the
67  * shared and distributed memory settings.
68  *
69  * \tparam VertexProgram The user defined vertex program which
70  * should implement the \ref graphlab::ivertex_program interface.
71  *
72  *
73  * ### Execution Semantics
74  *
75  * On start() the \ref graphlab::ivertex_program::init function is invoked
76  * on all vertex programs in parallel to initialize the vertex program,
77  * vertex data, and possibly signal vertices.
78  * The engine then proceeds to execute a sequence of
79  * super-steps (iterations) each of which is further decomposed into a
80  * sequence of minor-steps which are also executed synchronously:
81  * \li Receive all incoming messages (signals) by invoking the
82  * \ref graphlab::ivertex_program::init function on all
83  * vertex-programs that have incoming messages. If a
84  * vertex-program does not have any incoming messages then it is
85  * not active during this super-step.
86  * \li Execute all gathers for active vertex programs by invoking
87  * the user defined \ref graphlab::ivertex_program::gather function
88  * on the edge direction returned by the
89  * \ref graphlab::ivertex_program::gather_edges function. The gather
90  * functions can modify edge data but cannot modify the vertex
91  * program or vertex data and therefore can be executed on multiple
92  * edges in parallel. The gather type is used to accumulate (sum)
93  * the result of the gather function calls.
94  * \li Execute all apply functions for active vertex-programs by
95  * invoking the user defined \ref graphlab::ivertex_program::apply
96  * function passing the sum of the gather functions. If \ref
97  * graphlab::ivertex_program::gather_edges returns no edges then
98  * the default gather value is passed to apply. The apply function
99  * can modify the vertex program and vertex data.
100  * \li Execute all scatters for active vertex programs by invoking
101  * the user defined \ref graphlab::ivertex_program::scatter function
102  * on the edge direction returned by the
103  * \ref graphlab::ivertex_program::scatter_edges function. The scatter
104  * functions can modify edge data but cannot modify the vertex
105  * program or vertex data and therefore can be executed on multiple
106  * edges in parallel.
107  *
108  * ### Construction
109  *
110  * The synchronous engine is constructed by passing in a
111  * \ref graphlab::distributed_control object which manages coordination
112  * between engine threads and a \ref graphlab::distributed_graph object
113  * which is the graph on which the engine should be run. The graph should
114  * already be populated and cannot change after the engine is constructed.
115  * In the distributed setting all program instances (running on each machine)
116  * should construct an instance of the engine at the same time.
117  *
118  * Computation is initiated by signaling vertices using either
119  * \ref graphlab::synchronous_engine::signal or
120  * \ref graphlab::synchronous_engine::signal_all. In either case all
121  * machines should invoke signal or signal all at the same time. Finally,
122  * computation is initiated by calling the
123  * \ref graphlab::synchronous_engine::start function.
124  *
125  * ### Example Usage
126  *
127  * The following is a simple example demonstrating how to use the engine:
128  * \code
129  * #include <graphlab.hpp>
130  *
131  * struct vertex_data {
132  * // code
133  * };
134  * struct edge_data {
135  * // code
136  * };
137  * typedef graphlab::distributed_graph<vertex_data, edge_data> graph_type;
138  * typedef float gather_type;
139  * struct pagerank_vprog :
140  * public graphlab::ivertex_program<graph_type, gather_type> {
141  * // code
142  * };
143  *
144  * int main(int argc, char** argv) {
145  * // Initialize control plain using mpi
146  * graphlab::mpi_tools::init(argc, argv);
147  * graphlab::distributed_control dc;
148  * // Parse command line options
149  * graphlab::command_line_options clopts("PageRank algorithm.");
150  * std::string graph_dir;
151  * clopts.attach_option("graph", &graph_dir, graph_dir,
152  * "The graph file.");
153  * if(!clopts.parse(argc, argv)) {
154  * std::cout << "Error in parsing arguments." << std::endl;
155  * return EXIT_FAILURE;
156  * }
157  * graph_type graph(dc, clopts);
158  * graph.load_structure(graph_dir, "tsv");
159  * graph.finalize();
160  * std::cout << "#vertices: " << graph.num_vertices()
161  * << " #edges:" << graph.num_edges() << std::endl;
162  * graphlab::synchronous_engine<pagerank_vprog> engine(dc, graph, clopts);
163  * engine.signal_all();
164  * engine.start();
165  * std::cout << "Runtime: " << engine.elapsed_seconds();
166  * graphlab::mpi_tools::finalize();
167  * }
168  * \endcode
169  *
170  *
171  *
172  * <a name=engineopts>Engine Options</a>
173  * =====================
174  * The synchronous engine supports several engine options which can
175  * be set as command line arguments using \c --engine_opts :
176  *
177  * \li <b>max_iterations</b>: (default: infinity) The maximum number
178  * of iterations (super-steps) to run.
179  *
180  * \li <b>timeout</b>: (default: infinity) The maximum time in
181  * seconds that the engine may run. When the time runs out the
182  * current iteration is completed and then the engine terminates.
183  *
184  * \li <b>use_cache</b>: (default: false) This is used to enable
185  * caching. When caching is enabled the gather phase is skipped for
186  * vertices that already have a cached value. To use caching the
187  * vertex program must either clear (\ref icontext::clear_gather_cache)
188  * or update (\ref icontext::post_delta) the cache values of
189  * neighboring vertices during the scatter phase.
190  *
191  * \li \b snapshot_interval If set to a positive value, a snapshot
192  * is taken every this number of iterations. If set to 0, a snapshot
193  * is taken before the first iteration. If set to a negative value,
194  * no snapshots are taken. Defaults to -1. A snapshot is a binary
195  * dump of the graph.
196  *
197  * \li \b snapshot_path If snapshot_interval is set to a value >=0,
198  * this option must be specified and should contain a target basename
199  * for the snapshot. The path including folder and file prefix in
200  * which the snapshots should be saved.
201  *
202  * \see graphlab::omni_engine
203  * \see graphlab::async_consistent_engine
204  * \see graphlab::semi_synchronous_engine
205  */

788  * \endcode

1270  template<typename VertexProgram> execution_status::status_enum
1271  synchronous_engine<VertexProgram>::start() {
1272  if (vlocks.size() != graph.num_local_vertices())
1273  resize();
1274  completed_applys = 0;
1275  rmi.barrier();
1276 
1277  // Initialization code ==================================================
1278  // Reset event log counters?
1279  // Start the timer
1280  graphlab::timer timer; timer.start();
1281  start_time = timer::approx_time_seconds();
1282  iteration_counter = 0;
1283  force_abort = false;
1284  execution_status::status_enum termination_reason =
1285  execution_status::UNSET;
1286  // if (perform_init_vtx_program) {
1287  // // Initialize all vertex programs
1288  // run_synchronous( &synchronous_engine::initialize_vertex_programs );
1289  // }
1290  aggregator.start();
1291  rmi.barrier();
1292 
1293  if (snapshot_interval == 0) {
1294  graph.save_binary(snapshot_path);
1295  }
1296 
1297  float last_print = -5;
1298  if (rmi.procid() == 0) {
1299  logstream(LOG_EMPH) << "Iteration counter will only output every 5 seconds."
1300  << std::endl;
1301  }
1302  // Program Main loop ====================================================
1303  while(iteration_counter < max_iterations && !force_abort ) {
1304 
1305  // Check first to see if we are out of time
1306  if(timeout != 0 && timeout < elapsed_seconds()) {
1307  termination_reason = execution_status::TIMEOUT;
1308  break;
1309  }
1310 
1311  bool print_this_round = (elapsed_seconds() - last_print) >= 5;
1312 
1313  if(rmi.procid() == 0 && print_this_round) {
1314  logstream(LOG_EMPH)
1315  << rmi.procid() << ": Starting iteration: " << iteration_counter
1316  << std::endl;
1317  last_print = elapsed_seconds();
1318  }
1319  // Reset Active vertices ----------------------------------------------
1320  // Clear the active super-step and minor-step bits which will
1321  // be set upon receiving messages
1322  active_superstep.clear(); active_minorstep.clear();
1323  has_gather_accum.clear();
1324  rmi.barrier();
1325 
1326  // Exchange Messages --------------------------------------------------
1327  // Exchange any messages in the local message vectors
1328  // if (rmi.procid() == 0) std::cout << "Exchange messages..." << std::endl;
1329  run_synchronous( &synchronous_engine::exchange_messages );
1330  /**
1331  * Post conditions:
1332  * 1) only master vertices have messages
1333  */
1334 
1335  // Receive Messages ---------------------------------------------------
1336  // Receive messages to master vertices and then synchronize
1337  // vertex programs with mirrors if gather is required
1338  //
1339 
1340  // if (rmi.procid() == 0) std::cout << "Receive messages..." << std::endl;
1341  num_active_vertices = 0;
1342  run_synchronous( &synchronous_engine::receive_messages );
1343  if (sched_allv) {
1344  active_minorstep.fill();
1345  }
1346  has_message.clear();
1347  /**
1348  * Post conditions:
1349  * 1) there are no messages remaining
1350  * 2) All masters that received messages have their
1351  * active_superstep bit set
1352  * 3) All masters and mirrors that are to participate in the
1353  * next gather phases have their active_minorstep bit
1354  * set.
1355  * 4) num_active_vertices is the number of vertices that
1356  * received messages.
1357  */
1358 
1359  // Check termination condition ---------------------------------------
1360  size_t total_active_vertices = num_active_vertices;
1361  rmi.all_reduce(total_active_vertices);
1362  if (rmi.procid() == 0 && print_this_round)
1363  logstream(LOG_EMPH)
1364  << "\tActive vertices: " << total_active_vertices << std::endl;
1365  if(total_active_vertices == 0 ) {
1366  termination_reason = execution_status::TASK_DEPLETION;
1367  break;
1368  }
1369 
1370 
1371  // Execute gather operations-------------------------------------------
1372  // Execute the gather operation for all vertices that are active
1373  // in this minor-step (active-minorstep bit set).
1374  // if (rmi.procid() == 0) std::cout << "Gathering..." << std::endl;
1375  run_synchronous( &synchronous_engine::execute_gathers );
1376  // Clear the minor step bit since only super-step vertices
1377  // (only master vertices are required to participate in the
1378  // apply step)
1379  active_minorstep.clear(); // rmi.barrier();
1380  /**
1381  * Post conditions:
1382  * 1) gather_accum for all master vertices contains the
1383  * result of all the gathers (even if they are drawn from
1384  * cache)
1385  * 2) No minor-step bits are set
1386  */
1387 
1388  // Execute Apply Operations -------------------------------------------
1389  // Run the apply function on all active vertices
1390  // if (rmi.procid() == 0) std::cout << "Applying..." << std::endl;
1391  run_synchronous( &synchronous_engine::execute_applys );
1392  /**

1567  size_t vcount = 0;

1569  timer ti;
1570

1577  if (lvid_block_start >= graph.num_local_vertices()) break;
1578  // get the bit field from has_message
1579  size_t lvid_bit_block = active_minorstep.containing_word(lvid_block_start);
1580  if (lvid_bit_block == 0) continue;
1581  // initialize a word sized bitfield
1582  local_bitset.clear();
1583  local_bitset.initialize_from_mem(&lvid_bit_block, sizeof(size_t));
1584 
1585  foreach(size_t lvid_block_offset, local_bitset) {
1586  lvid_type lvid = lvid_block_start + lvid_block_offset;
1587  if (lvid >= graph.num_local_vertices()) break;
1588 
1589  bool accum_is_set = false;
1590  gather_type accum = gather_type();
1591  // if caching is enabled and we have a cache entry then use
1592  // that as the accum
1593  if( caching_enabled && has_cache.get(lvid) ) {
1594  accum = gather_cache[lvid];
1595  accum_is_set = true;
1596  } else {
1597  // recompute the local contribution to the gather
1598  const vertex_program_type& vprog = vertex_programs[lvid];
1599  local_vertex_type local_vertex = graph.l_vertex(lvid);
1600  const vertex_type vertex(local_vertex);
1601  const edge_dir_type gather_dir = vprog.gather_edges(context, vertex);
1602  // Loop over in edges
1603  size_t edges_touched = 0;
1604  vprog.pre_local_gather(accum);
1605  if(gather_dir == IN_EDGES || gather_dir == ALL_EDGES) {
1606  foreach(local_edge_type local_edge, local_vertex.in_edges()) {
1607  edge_type edge(local_edge);
1608  // elocks[local_edge.id()].lock();
1609  if(accum_is_set) { // \todo hint likely
1610  accum += vprog.gather(context, vertex, edge);
1611  } else {
1612  accum = vprog.gather(context, vertex, edge);
1613  accum_is_set = true;
1614  }
1615  ++edges_touched;
1616  // elocks[local_edge.id()].unlock();
1617  }
1618  } // end of if in_edges/all_edges
1619  // Loop over out edges

1622  edge_type edge(local_edge);
1623  // elocks[local_edge.id()].lock();
1624  if(accum_is_set) { // \todo hint likely
1625  accum += vprog.gather(context, vertex, edge);
1626  } else {
1627  accum = vprog.gather(context, vertex, edge);
1628  accum_is_set = true;
1629  }
1630  // elocks[local_edge.id()].unlock();
1631  ++edges_touched;
1632  }
1633  INCREMENT_EVENT(EVENT_GATHERS, edges_touched);
1634  } // end of if out_edges/all_edges
1635  vprog.post_local_gather(accum);
1636  // If caching is enabled then save the accumulator to the
1637  // cache for future iterations. Note that it is possible
1638  // that the accumulator was never set in which case we are
1639  // effectively "zeroing out" the cache.
1640  if(caching_enabled && accum_is_set) {
1641  gather_cache[lvid] = accum; has_cache.set_bit(lvid);
1642  } // end of if caching enabled
1643  }
1644  // If the accum contains a value for the local gather we put
1645  // that estimate in the gather exchange.
1646  if(accum_is_set) sync_gather(lvid, accum, thread_id);
1647  if(!graph.l_is_master(lvid)) {
1648  // if this is not the master clear the vertex program
1649  vertex_programs[lvid] = vertex_program_type();
1650  }
1651 
1652  // try to recv gathers if there are any in the buffer
1653  if(++vcount % TRY_RECV_MOD == 0) recv_gathers();
1654  }
1655  } // end of loop over vertices to compute gather accumulators
1656  per_thread_compute_time[thread_id] += ti.current_time();
1657  gather_exchange.partial_flush();
1658  // Finish sending and receiving all gather operations
1659  thread_barrier.wait();
1660  if(thread_id == 0) gather_exchange.flush();
1661  thread_barrier.wait();
1662  recv_gathers();
1663  } // end of execute_gathers
