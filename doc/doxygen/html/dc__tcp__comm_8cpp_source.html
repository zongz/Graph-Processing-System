62  void dc_tcp_comm::init(const std::vector<std::string> &machines,
63  const std::map<std::string,std::string> &initopts,
64  procid_t curmachineid,
65  std::vector<dc_receive*> receiver_,
66  std::vector<dc_send*> sender_) {
67 
68  curid = curmachineid;
69  ASSERT_LT(machines.size(), std::numeric_limits<procid_t>::max());
70  nprocs = (procid_t)(machines.size());
71  receiver = receiver_;
72  sender = sender_;
73 
74  // insert machines into the address map
75  all_addrs.resize(nprocs);
76  portnums.resize(nprocs);
77  assert(triggered_timeouts.size() >= nprocs);
78  triggered_timeouts.clear();
79  // fill all the socks
80  sock.resize(nprocs);
81  for (size_t i = 0;i < nprocs; ++i) {
82  sock[i].id = i;
83  sock[i].owner = this;
84  sock[i].outsock = -1;
85  sock[i].insock = -1;
86  sock[i].inevent = NULL;
87  sock[i].outevent = NULL;
88  sock[i].wouldblock = false;
89  sock[i].data.msg_name = NULL;
90  sock[i].data.msg_namelen = 0;
91  sock[i].data.msg_control = NULL;
92  sock[i].data.msg_controllen = 0;
93  sock[i].data.msg_flags = 0;
94  sock[i].data.msg_iovlen = 0;
95  sock[i].data.msg_iov = NULL;
96  }
97 
98  program_md5 = get_current_process_hash();
99  ASSERT_EQ(program_md5.length(), 32);
100  // parse the machines list, and extract the relevant address information
101  for (size_t i = 0;i < machines.size(); ++i) {
102  // extract the port number
103  size_t pos = machines[i].find(":");
104  ASSERT_NE(pos, std::string::npos);
105  std::string address = machines[i].substr(0, pos);
106  size_t port = boost::lexical_cast<size_t>(machines[i].substr(pos+1));
107 
108  struct hostent* ent = gethostbyname(address.c_str());
109  ASSERT_EQ(ent->h_length, 4);
110  uint32_t addr = *reinterpret_cast<uint32_t*>(ent->h_addr_list[0]);
111 
112  all_addrs[i] = addr;
113  ASSERT_LT(port, 65536);
114  portnums[i] = (uint16_t)(port);
115  }
116  network_bytessent = 0;
117  buffered_len = 0;
118  // if sock handle is set
119  std::map<std::string, std::string>::const_iterator iter =
120  initopts.find("__sockhandle__");
121  if (iter != initopts.end()) {
122  open_listening(atoi(iter->second.c_str()));
123  } else {
124  open_listening();
125  }
126  // to improve the "synchronous" nature of the connection setup,
127  // the last machine will do this in reverse order.
128  // To wait for all machines to connect to it, before it
129  // tries to connect to all machines.
130  // The last machine therefore essentially acts as the "barrier" leader
131  if (curid != nprocs - 1) {
132  // not the last machine.
133  // Connect to everyone, EXCEPT the last machine
134  // and wait for all incoming connections
135  for(procid_t i = 0;i < nprocs - 1; ++i) connect(i);
136 
137  // wait for p - 1 incoming connections
138  insock_lock.lock();
139  while(1) {
140  if (num_in_connected() == sock.size() - 1) break;
141  insock_cond.wait(insock_lock);
142  }
143  insock_lock.unlock();
144 
145  // connect to the last machine
146  connect(nprocs - 1);
147 
148  // wait for the last connection
149  insock_lock.lock();
150  while(1) {
151  if (num_in_connected() == sock.size()) break;
152  insock_cond.wait(insock_lock);
153  }
154  insock_lock.unlock();
155  } else {
156  // I am the last machine
157  ASSERT_EQ(curid, nprocs - 1);
158  // wait for all incoming connections before connecting to everyone
159  // connect to myself
160  connect(nprocs - 1);
161  insock_lock.lock();
162  while(1) {
163  if (num_in_connected() == sock.size()) break;
164  insock_cond.wait(insock_lock);
165  }
166  insock_lock.unlock();
167  // now, when I know that machines 0 to #procs - 2 have
168  // all established a connection to each other
169  // connect to everyone. This is essentially equivalent to the
170  // barrier release message
171  for(size_t i = 0;i < nprocs; ++i) connect(i);
172  }
173  // everyone is connected.
174  // Construct the eventbase
175  construct_events();
176  // we reserve the last 2 cores for communication
177  inthreads.launch(boost::bind(&dc_tcp_comm::receive_loop, this, inevbase), thread::cpu_count() - 2);
178  outthreads.launch(boost::bind(&dc_tcp_comm::send_loop, this, outevbase), thread::cpu_count() - 1);
179  is_closed = false;
180  }

182  void dc_tcp_comm::construct_events() {
183  int ret = evthread_use_pthreads();
184  if (ret < 0) logstream(LOG_FATAL) << "Unable to initialize libevent with pthread support!" << std::endl;
185  // number of evs to create.
186  outevbase = event_base_new();
187  if (!outevbase) logstream(LOG_FATAL) << "Unable to construct libevent base" << std::endl;
188  send_all_timeout.owner = this;
189  send_all_timeout.send_all = true;
190  send_triggered_timeout.owner = this;
191  send_triggered_timeout.send_all = false;
192  send_all_event = event_new(outevbase, -1, EV_TIMEOUT | EV_PERSIST, on_send_event, &(send_all_timeout));
193  assert(send_all_event != NULL);
194  struct timeval t = {SEND_POLL_TIMEOUT / 1000000, SEND_POLL_TIMEOUT % 1000000} ;
195  event_add(send_all_event, &t);
196  send_triggered_event = event_new(outevbase, -1, EV_TIMEOUT | EV_PERSIST, on_send_event, &(send_triggered_timeout));
197  assert(send_triggered_event != NULL);
198 
199  inevbase = event_base_new();
200  if (!inevbase) logstream(LOG_FATAL) << "Unable to construct libevent base" << std::endl;
201 
202 
203  //register all event objects
204  for (size_t i = 0;i < sock.size(); ++i) {
205  sock[i].inevent = event_new(inevbase, sock[i].insock, EV_READ | EV_PERSIST | EV_ET,
206  on_receive_event, &(sock[i]));
207  if (sock[i].inevent == NULL) {
208  logstream(LOG_FATAL) << "Unable to register socket read event" << std::endl;
209  }
210 
211  sock[i].outevent = event_new(outevbase, sock[i].outsock, EV_WRITE | EV_PERSIST | EV_ET,
212  on_send_event, &(sock[i]));
213  if (sock[i].outevent == NULL) {
214  logstream(LOG_FATAL) << "Unable to register socket write event" << std::endl;
215  }
216 
217  event_add(sock[i].inevent, NULL);
218  //struct timeval t = {0, 10};
219  event_add(sock[i].outevent, NULL);
220  }
221  }

248  // close the listening socket
249  if (listensock > 0) {
250  ::close(listensock);

253  // shutdown the listening thread
254  listenthread.join();
255 
256  // clear the outevent loop
257  event_base_loopbreak(outevbase);
258  outthreads.join();
259  for (size_t i = 0;i < sock.size(); ++i) {
260  event_free(sock[i].outevent);
261  }
262  event_free(send_triggered_event);
263  event_free(send_all_event);
264  event_base_free(outevbase);
265 
266 
267  logstream(LOG_INFO) << "Closing outgoing sockets" << std::endl;
268  // close all outgoing sockets
269  for (size_t i = 0;i < sock.size(); ++i) {
270  if (sock[i].outsock > 0) {
271  ::close(sock[i].outsock);
272  sock[i].outsock = -1;
273  }
274  }
275 
276  // clear the inevent loop
277  event_base_loopbreak(inevbase);
278  inthreads.join();
279  for (size_t i = 0;i < sock.size(); ++i) {
280  event_free(sock[i].inevent);
281  }
282  event_base_free(inevbase);
283 
284 
285  logstream(LOG_INFO) << "Closing incoming sockets" << std::endl;
286  // close all incoming sockets

297  bool dc_tcp_comm::send_till_block(socket_info& sockinfo) {
298  sockinfo.wouldblock = false;
299  // while there is still data to be sent
300  BEGIN_TRACEPOINT(tcp_send_call);
301  while(!sockinfo.outvec.empty()) {
302  sockinfo.outvec.fill_msghdr(sockinfo.data);
303  ssize_t ret = sendmsg(sockinfo.outsock, &sockinfo.data, 0);
304  if (ret < 0) {
305  END_TRACEPOINT(tcp_send_call);
306  if (errno == EWOULDBLOCK || errno == EAGAIN) {
307  sockinfo.wouldblock = true;
308  return false;
309  }
310  else {
311  logstream(LOG_FATAL) << "send error: " << strerror(errno) << std::endl;
312  return false;
313  }
314  }
315 
316 #ifdef COMM_DEBUG
317  logstream(LOG_INFO) << ret << " bytes --> " << sockinfo.id << std::endl;
318 #endif
319  network_bytessent.inc(ret);
320  sockinfo.outvec.sent(ret);
321  }
322  END_TRACEPOINT(tcp_send_call);
323  return true;
324  }

326  int dc_tcp_comm::sendtosock(int sockfd, const char* buf, size_t len) {
327  size_t numsent = 0;
328  BEGIN_TRACEPOINT(tcp_send_call);
329  while (numsent < len) {
330  ssize_t ret = ::send(sockfd, buf + numsent, len - numsent, 0);
331  if (ret < 0) {
332  logstream(LOG_ERROR) << "send error: " << strerror(errno) << std::endl;
333  END_TRACEPOINT(tcp_send_call);
334  return errno;
335  }
336  numsent += ret;
337  }
338  END_TRACEPOINT(tcp_send_call);
339  return 0;
340  }

390  void dc_tcp_comm::open_listening(int sockhandle) {
391  // open listening socket
392  if (sockhandle == 0) {
393  listensock = socket(AF_INET, SOCK_STREAM, 0);
394  // uninteresting boiler plate. Set the port number and socket type
395  sockaddr_in my_addr;
396  my_addr.sin_family = AF_INET;
397  my_addr.sin_port = htons(portnums[curid]);
398  my_addr.sin_addr.s_addr = INADDR_ANY;
399  memset(&(my_addr.sin_zero), '\0', 8);
400  logstream(LOG_INFO) << "Proc " << procid() << " Bind on "
401  << portnums[curid] << "\n";
402  if (bind(listensock, (sockaddr*)&my_addr, sizeof(my_addr)) < 0)
403  {
404  logstream(LOG_FATAL) << "bind: " << strerror(errno) << "\n";
405  ASSERT_TRUE(0);
406  }
407  }
408  else {
409  listensock = sockhandle;
410  }
411  logstream(LOG_INFO) << "Proc " << procid()
412  << " listening on " << portnums[curid] << "\n";
413  ASSERT_EQ(0, listen(listensock, 128));
414  // spawn a thread which loops around accept
415  listenthread.launch(boost::bind(&dc_tcp_comm::accept_handler, this));

418  void dc_tcp_comm::connect(size_t target) {
419  if (sock[target].outsock != -1) {
420  return;
421  } else {
422  int newsock = socket(AF_INET, SOCK_STREAM, 0);
423  set_tcp_no_delay(newsock);
424  sockaddr_in serv_addr;
425  serv_addr.sin_family = AF_INET;
426  // set the target port
427  serv_addr.sin_port = htons(portnums[target]);
428  // set the target address
429  serv_addr.sin_addr = *(struct in_addr*)&(all_addrs[target]);
430  memset(&(serv_addr.sin_zero), '\0', 8);
431  // Connect!
432  logstream(LOG_INFO) << "Trying to connect from "
433  << curid << " -> " << target
434  << " on port " << portnums[target] << "\n";
435  logger(LOG_INFO, "Destination IP = %s", inet_ntoa(serv_addr.sin_addr));
436  // retry 10 times at 1 second intervals
437  bool success = false;
438  for (size_t i = 0;i < 10; ++i) {
439  if (::connect(newsock, (sockaddr*)&serv_addr, sizeof(serv_addr)) < 0) {
440  logstream(LOG_INFO)
441  << "connect " << curid << " to " << target << ": "
442  << strerror(errno) << ". Retrying...\n";
443  timer::sleep(1);
444  // posix says that
445  /* If connect() fails, the state of the socket is unspecified.
446  Conforming applications should close the file descriptor and
447  create a new socket before attempting to reconnect. */
448  ::close(newsock);
449  newsock = socket(AF_INET, SOCK_STREAM, 0);
450  set_tcp_no_delay(newsock);
451  } else {
452  // send the initial message
453  initial_message msg;
454  msg.id = curid;
455  memcpy(msg.md5, program_md5.c_str(), 32);
456  sendtosock(newsock, reinterpret_cast<char*>(&msg), sizeof(initial_message));
457  set_non_blocking(newsock);
458  success = true;
459  break;
460  }
461  }
462  if (!success) {
463  logstream(LOG_FATAL) << "Failed to establish connection" << std::endl;
464  }
465  // remember the socket
466  sock[target].outsock = newsock;
467  logstream(LOG_INFO) << "connection from " << curid << " to " << target
468  << " established." << std::endl;
469  }
470  } // end of connect

480  // waits for incoming connections
481  void dc_tcp_comm::accept_handler() {
482  pollfd pf;
483  pf.fd = listensock;
484  pf.events = POLLIN;
485  pf.revents = 0;
486  size_t numsocks_connected = 0;
487  logstream(LOG_INFO) << "Listening thread launched." << std::endl;
488  while(numsocks_connected < sock.size()) {
489  // wait for incoming event
490  poll(&pf, 1, 1000);
491  // if we have a POLLIN, we have an incoming socket request
492  if (pf.revents & POLLIN) {
493  logstream(LOG_INFO) << "Accepting...." << std::endl;
494  // accept the socket
495  sockaddr_in their_addr;
496  socklen_t namelen = sizeof(sockaddr_in);
497  int newsock = accept(listensock, (sockaddr*)&their_addr, &namelen);
498  logstream(LOG_INFO) << "Accepted" << std::endl;
499  if (newsock < 0) {
500  break;
501  }
502  // set the socket options and inform the
503  set_tcp_no_delay(newsock);
504  // before accepting the socket, get the machine number
505  initial_message remote_message;
506  ssize_t msglen = 0;
507  while(msglen != sizeof(initial_message)) {
508  int retval = recv(newsock, (char*)(&remote_message) + msglen,
509  sizeof(initial_message) - msglen, 0);
510  if (retval < 0) {
511  if (errno == EWOULDBLOCK || errno == EAGAIN) {
512  continue;
513  }
514  else {
515  logstream(LOG_FATAL) << "error: " << errno << " receive error: " << strerror(errno) << std::endl;
516  }
517  }
518  else if (retval > 0) {
519  msglen += retval;
520  }
521  else if (retval == 0) {
522  std::cout << "error: connection dropped." << std::endl;
523  ::close(newsock);
524  newsock = -1;
525  break;
526  }
527  }
528  if (newsock != -1) {
529  // validate the md5 hash
530  std::string other_md5 = std::string(remote_message.md5, 32);
531  if (other_md5 != program_md5) {
532  logstream(LOG_FATAL) << "MD5 mismatch. \n "
533  << "\tProcess " << curid << " has hash " << program_md5 << " \n "
534  << "\tProcess " << remote_message.id << " has hash " << other_md5 << " \n "
535  << "\tGraphLab requires all machines to run exactly the same binary." << std::endl;
536  }
537  // register the new socket
538  set_non_blocking(newsock);
539  new_socket(newsock, &their_addr, remote_message.id);
540  ++numsocks_connected;
541  }
542  }
543  if (listensock == -1) {
544  // the owner has closed
545  break;
546  }
547  }
548  logstream(LOG_INFO) << "Listening thread quitting" << std::endl;
549  } // end of run

552  // libevent receive handler
553  void on_receive_event(int fd, short ev, void* arg) {
554  dc_tcp_comm::socket_info* sockinfo = (dc_tcp_comm::socket_info*)(arg);
555  dc_tcp_comm* comm = sockinfo->owner;
556  if (ev & EV_READ) {
557  // get a direct pointer to my receiver
558  dc_receive* receiver = comm->receiver[sockinfo->id];
559 
560  size_t buflength;
561  char *c = receiver->get_buffer(buflength);
562  while(1) {
563  ssize_t msglen = recv(fd, c, buflength, 0);
564  if (msglen < 0) {
565  if (errno == EAGAIN || errno == EWOULDBLOCK) break;
566  else {
567  logstream(LOG_FATAL) << "receive error: " << strerror(errno) << std::endl;
568  break;
569  }
570  }
571  else if (msglen == 0) {
572  // socket closed
573  break;
574  }
575  else if (msglen > 0) {
576  comm->network_bytesreceived.inc(msglen);
577  #ifdef COMM_DEBUG
578  logstream(LOG_INFO) << msglen << " bytes <-- "
579  << sockinfo->id << std::endl;
580  #endif
581  c = receiver->advance_buffer(c, msglen, buflength);
582  }
583  }
584  }
585  }
