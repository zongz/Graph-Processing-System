62  template <typename Graph>
63  bool snap_parser(Graph& graph, const std::string& srcfilename,
64  const std::string& str) {
65  if (str.empty()) return true;
66  else if (str[0] == '#') {
67  std::cout << str << std::endl;
68  } else {
69  size_t source, target;
70  char* targetptr;
71  source = strtoul(str.c_str(), &targetptr, 10);
72  if (targetptr == NULL) return false;
73  target = strtoul(targetptr, NULL, 10);
74  if(source != target) graph.add_edge(source, target);
75  }
76  return true;
77  } // end of snap parser

141  return true;

146  template <typename Graph>
147  bool adj_parser(Graph& graph, const std::string& srcfilename,
148  const std::string& line) {
149  // If the line is empty simply skip it
150  if(line.empty()) return true;
151  // We use the boost spirit parser which requires (too) many separate
152  // namespaces so to make things clear we shorten them here.
153  namespace qi = boost::spirit::qi;
154  namespace ascii = boost::spirit::ascii;
155  namespace phoenix = boost::phoenix;
156  vertex_id_type source(-1);
157  vertex_id_type ntargets(-1);
158  std::vector<vertex_id_type> targets;
159  const bool success = qi::phrase_parse
160  (line.begin(), line.end(),
161  // Begin grammar
162  (
163  qi::ulong_[phoenix::ref(source) = qi::_1] >> -qi::char_(",") >>
164  qi::ulong_[phoenix::ref(ntargets) = qi::_1] >> -qi::char_(",") >>
165  *(qi::ulong_[phoenix::push_back(phoenix::ref(targets), qi::_1)] % -qi::char_(","))
166  )
167  ,
168  // End grammar
169  ascii::space);
170  // Test to see if the boost parser was able to parse the line
171  if(!success || ntargets != targets.size()) {
172  logstream(LOG_ERROR) << "Parse error in vertex prior parser." << std::endl;
173  return false;
174  }
175  for(size_t i = 0; i < targets.size(); ++i) {
176  if(source != targets[i]) graph.add_edge(source, targets[i]);
177  }
178  return true;
179  } // end of adj parser

200  /**
201  * \internal
202  * Replaces \255 with \255\1
203  * Replaces \\n with \255\0
204  */
205  static std::string escape_newline(charstream& strm) {
206  size_t ctr = 0;
207  char *ptr = strm->c_str();
208  size_t strmlen = strm->size();
209  for (size_t i = 0;i < strmlen; ++i) {
210  ctr += (ptr[i] == (char)255 || ptr[i] == '\n');
211  }
212 
213  std::string ret(ctr + strmlen, 0);
214 
215  size_t target = 0;
216  for (size_t i = 0;i < strmlen; ++i, ++ptr) {
217  if ((*ptr) == (char)255) {
218  ret[target] = (char)255;
219  ret[target + 1] = 1;
220  target += 2;
221  }
222  else if ((*ptr) == '\n') {
223  ret[target] = (char)255;
224  ret[target + 1] = 0;
225  target += 2;
226  }
227  else {
228  ret[target] = (*ptr);
229  ++target;
230  }
231  }
232  assert(target == ctr + strmlen);
233  return ret;
234  }

236  /**
237  * \internal
238  * Replaces \255\1 with \255
239  * Replaces \255\0 with \\n
240  */
241  static std::string unescape_newline(const std::string& str) {
242  size_t ctr = 0;
243  // count the number of escapes
244  for (size_t i = 0;i < str.length(); ++i) {
245  ctr += (str[i] == (char)255);
246  }
247  // real length is string length - escapes
248  std::string ret(str.size() - ctr, 0);
249 
250  const char escapemap[2] = {'\n', (char)255};
251 
252  size_t target = 0;
253  for (size_t i = 0;i < str.length(); ++i, ++target) {
254  if (str[i] == (char)255) {
255  // escape character
256  ++i;
257  ASSERT_MSG(str[i] == 0 || str[i] == 1,
258  "Malformed escape sequence in graphjrl file.");
259  ret[target] = escapemap[(int)str[i]];
260  }
261  else {
262  ret[target] = str[i];
263  }
264  }
265  return ret;
266  }

287  template <typename Graph>
288  bool graphjrl_parser(Graph& graph, const std::string& srcfilename,
289  const std::string& str) {
290  std::string unescapedstr = graphjrl_writer<Graph>::unescape_newline(str);
291  boost::iostreams::stream<boost::iostreams::array_source>
292  istrm(unescapedstr.c_str(), unescapedstr.length());
293  iarchive iarc(istrm);
294 
295  char entrytype;
296  iarc >> entrytype;
297  if (entrytype == 0) {
298  typename Graph::vertex_id_type vid;
299  typename Graph::vertex_data_type vdata;
300  iarc >> vid >> vdata;
301  graph.add_vertex(vid, vdata);
302  }
303  else if (entrytype == 1) {
304  typename Graph::vertex_id_type srcvid, destvid;
305  typename Graph::edge_data_type edata;
306  iarc >> srcvid >> destvid >> edata;
307  graph.add_edge(srcvid, destvid, edata);
308  }
309  else {
310  return false;
311  }
312  return true;
313  }
