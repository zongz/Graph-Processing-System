38  * Actual Parfor implementation.
39  * Holds a reference to all the arguments.
40  * Each fiber increments the atomic counter and runs the fn on it)

42 template <typename GraphType>
43 struct parfor_all_vertices_impl{
44 
45  GraphType& graph;
46  boost::function<void(typename GraphType::vertex_type)> fn;
47  vertex_set& vset;
48  atomic<size_t> ctr;
49 
50  parfor_all_vertices_impl(GraphType& graph,
51  boost::function<void(typename GraphType::vertex_type)> fn,
52  vertex_set& vset): graph(graph),fn(fn),vset(vset),ctr(0) { }

54  void run_fiber() {
55  while (1) {
56  size_t lvid = ctr.inc_ret_last();
57  if (lvid >= graph.num_local_vertices() || !vset.l_contains(lvid)) break;
58  typename GraphType::local_vertex_type l_vertex = graph.l_vertex(lvid);
59  if (l_vertex.owned()) {
60  typename GraphType::vertex_type vertex(l_vertex);
61  fn(vertex);
62  }
63  }
64  }
65 };

70 /**
71  * \ingroup warp
72  *
73  * This Warp function provides a simple parallel for loop over all vertices
74  * in the graph, or in a given set of vertices. A large number of light-weight
75  * threads called fibers are used to run the user specified function,
76  * allowing the user to make what normally will be blocking calls, on
77  * the neighborhood of each vertex.
78  *
79  * \code
80  * float pagerank_map(graph_type::edge_type edge, graph_type::vertex_type other) {
81  * return other.data() / other.num_out_edges();
82  * }
83  *
84  * void pagerank(graph_type::vertex_type vertex) {
85  * // computes an aggregate over the neighborhood using map_reduce_neighborhood
86  * vertex.data() = 0.15 + 0.85 * warp::map_reduce_neighborhood(vertex,
87  * IN_EDGES,
88  * pagerank_map);
89  * }
90  *
91  * ...
92  * // runs the pagerank function on all the vertices in the graph.
93  * parfor_all_vertices(graph, pagerank);
94  * \endcode
95  *
96  * \param graph A reference to the graph object
97  * \param fn A function to run on each vertex. Has the prototype void(GraphType::vertex_type). Can be a boost::function
98  * \param vset A set of vertices to run on
99  * \param nfibers Number of fiber threads to use. Defaults to 10000
100  * \param stacksize Size of each fiber stack in bytes. Defaults to 16384 bytes
101  *
102  * \see graphlab::warp::map_reduce_neighborhood()
103  * \see graphlab::warp::warp_graph_transform()
104  */
105 template <typename GraphType, typename FunctionType>
106 void parfor_all_vertices(GraphType& graph,
107  FunctionType fn,
108  vertex_set vset = GraphType::complete_set(),
109  size_t nfibers = 10000,
110  size_t stacksize = 16384) {
111  distributed_control::get_instance()->barrier();
112  bool old_fast_track = distributed_control::get_instance()->set_fast_track_requests(false);
113  fiber_group group;
114  group.set_stacksize(stacksize);
115  warp_impl::parfor_all_vertices_impl<GraphType> parfor(graph, fn, vset);
116 
117  for (size_t i = 0;i < nfibers; ++i) {
118  group.launch(boost::bind(&warp_impl::parfor_all_vertices_impl<GraphType>::run_fiber, &parfor));
119  }
120  group.join();
121  distributed_control::get_instance()->barrier();
122  distributed_control::get_instance()->set_fast_track_requests(old_fast_track);
123  graph.synchronize(vset);
124 }
