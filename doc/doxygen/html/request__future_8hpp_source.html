35  /**
36  * \ingroup rpc
37  * The result of a remote_request future call.
38  * This class represents the outcome of a remote request sent to another
39  * machine via the future-based remote_request_call. The future remote_request call
40  * returns immediately with this object. Only when operator() is called on this
41  * object, then it waits for a result from the remote machine. All remote_request
42  * calls which return futures are linked below.
43  *
44  * example:
45  * \code
46  * // this function returns immediately
47  * graphlab::request_future<int> res =
48  * rmi.future_remote_request(SOME_OTHER_MACHINE,
49  * function_which_returns_an_integer, ...);
50  *
51  * ... we can do other stuff ...
52  * // read the result, or wait for the result if it is not done yet.
53  * int actual_result = res();
54  * \endcode
55  *
56  * \see graphlab::distributed_control::future_remote_request
57  * graphlab::dc_dist_object::future_remote_request
58  * graphlab::fiber_remote_request
59  * graphlab::object_fiber_remote_request
60  *
61  * The future object holds a copy of the result of the request, and the
62  * operator() call returns a reference to this result (once it is available).
63  */
64 template <typename T>
65 struct request_future {
66  typedef typename dc_impl::function_ret_type<T>::type result_type;
67  mutable std::auto_ptr<dc_impl::ireply_container> reply;
68  result_type result;
69  bool hasval;
70 
71  /// default constructor
72  request_future():
73  reply(new dc_impl::basic_reply_container),
74  hasval(false) { }
75 
76 
77  /** constructor which allows you to specify a custom target container
78  * This class takes ownership of the container and will free it when done.
79  */
80  request_future(dc_impl::ireply_container* container):
81  reply(container),
82  hasval(false) { }

85  /** We can assign return values directly to the future in the
86  * case where no remote calls are necessary.
87  * Thus allowing the following to be written easily:
88  * \code
89  * request_future<int> a_function(int arg) {
90  * if (arg == 0) return rmi.future_remote_request(... somewhere else ...) ;
91  * else return 10;
92  * }
93  * \endcode
94  */
95  request_future(const T& val):
96  reply(NULL),
97  result(val),
98  hasval(true) { }
99 
100  /// copy constructor
101  request_future(const request_future<T>& val):
102  reply(val.reply),
103  result(val.result),
104  hasval(val.hasval) { }
105 
106  /// operator=
107  request_future& operator=(const request_future<T>& val) {
108  reply = val.reply;
109  result = val.result;
110  hasval = val.hasval;
111  return *this;
112  }
113 
114  /**
115  * \internal
116  * Returns a handle to the underlying container
117  */
118  size_t get_handle() {
119  return reinterpret_cast<size_t>(reply.get());
120  }
121 
122  /**
123  * Waits for the request if it has not yet been received.
124  */
125  void wait() {
126  if (!hasval) {
127  reply->wait();
128  dc_impl::blob& receiveddata = reply->get_blob();
129  iarchive iarc(receiveddata.c, receiveddata.len);
130  iarc >> result;
131  receiveddata.free();
132  hasval = true;
133  }
134  }
135 
136  /**
137  * Returns true if the result is ready and \ref operator()
138  * can be called without blocking.
139  */
140  bool is_ready() {
141  return (hasval || reply->ready());
142  }
143 
144  /**
145  * Waits for the request if it has not yet been received.
146  * When the result is ready, it returns a reference to the received value.
147  */
148  result_type& operator()() {
149  if (!hasval) wait();
150  return result;
151  }
152 };
