61  new_scheduler(size_t num_vertices, const graphlab_options& opts) {
62  std::string scheduler_str = opts.get_scheduler_type();
63 #define __GENERATE_NEW_SCHEDULER__(r_unused, data_unused, i, elem) \
64  BOOST_PP_EXPR_IF(i, else) \
65  if (scheduler_str == BOOST_PP_TUPLE_ELEM(3,0,elem)) { \
66  typedef BOOST_PP_TUPLE_ELEM(3,1,elem) \
67  scheduler_type; \
68  return new_scheduler_impl<scheduler_type> \
69  ( num_vertices, opts); \
70  }
71  // generate the construction calls
72  BOOST_PP_SEQ_FOR_EACH_I(__GENERATE_NEW_SCHEDULER__, _, __SCHEDULER_LIST__);
73 #undef __GENERATE_NEW_SCHEDULER__
74  logstream(LOG_FATAL)
75  << "Invalid scheduler type: " << scheduler_str << std::endl;
76  return NULL;
77  } // end of new_scheduler
