263  /// Like wait() but with a time limit of "ms" milliseconds
264  inline int timedwait_ms(const mutex& mut, size_t ms) const {
265  struct timespec timeout;
266  struct timeval tv;
267  gettimeofday(&tv, NULL);
268  // convert ms to s and ns
269  size_t s = ms / 1000;
270  ms = ms % 1000;
271  size_t ns = ms * 1000000;
272  // convert timeval to timespec
273  timeout.tv_nsec = tv.tv_usec * 1000;
274  timeout.tv_sec = tv.tv_sec;
275 
276  // add the time
277  timeout.tv_nsec += (suseconds_t)ns;
278  timeout.tv_sec += (time_t)s;
279  // shift the nsec to sec if overflow
280  if (timeout.tv_nsec > 1000000000) {
281  timeout.tv_sec ++;
282  timeout.tv_nsec -= 1000000000;
283  }
284  return pthread_cond_timedwait(&m_cond, &mut.m_mut, &timeout);
285  }

502  struct spinrwlock2 {
503  mutable unsigned int l;
504 
505  spinrwlock2():l(0) {}
506  void writelock() const {
507  while (1) {
508  unsigned state = l;
509 
510  /* No readers or writers? */
511  if (state < RW_WRITE)
512  {
513  /* Turn off RW_WAIT, and turn on RW_WRITE */
514  if (cmpxchg(&l, state, RW_WRITE) == state) return;
515 
516  /* Someone else got there... time to wait */
517  state = l;
518  }
519 
520  /* Turn on writer wait bit */
521  if (!(state & RW_WAIT)) atomic_set_bit(&l, RW_WAIT_BIT);
522 
523  /* Wait until can try to take the lock */
524  while (l > RW_WAIT) cpu_relax();
525  }
526  }
527 
528  void wrunlock() const {
529  atomic_add(&l, -RW_WRITE);
530  }
531 
532  void readlock() const {
533  while (1) {
534  /* A writer exists? */
535  while (l & (RW_WAIT | RW_WRITE)) cpu_relax();
536 
537  /* Try to get read lock */
538  if (!(atomic_xadd(&l, RW_READ) & (RW_WAIT | RW_WRITE))) return;
539 
540  /* Undo */
541  atomic_add(&l, -RW_READ);
542  }
543  }
544 
545  void rdunlock() const {
546  atomic_add(&l, -RW_READ);
547  }
548  };

627  /**
628  * \ingroup util
629  * This is a simple sense-reversing barrier implementation.
630  * In addition to standard barrier functionality, this also
631  * provides a "cancel" function which can be used to destroy
632  * the barrier, releasing all threads stuck in the barrier.
633  *
634  * Before you use, see \ref parallel_object_intricacies.
635  */
636  class cancellable_barrier {
637  private:
638  graphlab::mutex mutex;
639  graphlab::conditional conditional;
640  mutable int needed;
641  mutable int called;
642 
643  mutable bool barrier_sense;
644  mutable bool barrier_release;
645  bool alive;
646 
647  // not copyconstructible
648  cancellable_barrier(const cancellable_barrier&) { }
649 
650 
651  public:
652  /// Construct a barrier which will only fall when numthreads enter
653  cancellable_barrier(size_t numthreads) {
654  needed = numthreads;
655  called = 0;
656  barrier_sense = false;
657  barrier_release = true;
658  alive = true;
659  }
660 
661  // not copyable
662  void operator=(const cancellable_barrier& m) { }
663 
664  void resize_unsafe(size_t numthreads) {
665  needed = numthreads;
666  }
667 
668  /**
669  * \warning: This barrier is safely NOT reusable with this cancel
670  * definition
671  */
672  inline void cancel() {
673  alive = false;
674  conditional.broadcast();
675  }
676  /// Wait on the barrier until numthreads has called wait
677  inline void wait() const {
678  if (!alive) return;
679  mutex.lock();
680  // set waiting;
681  called++;
682  bool listening_on = barrier_sense;
683  if (called == needed) {
684  // if I have reached the required limit, wait up. Set waiting
685  // to 0 to make sure everyone wakes up
686  called = 0;
687  barrier_release = barrier_sense;
688  barrier_sense = !barrier_sense;
689  // clear all waiting
690  conditional.broadcast();
691  } else {
692  // while no one has broadcasted, sleep
693  while(barrier_release != listening_on && alive) conditional.wait(mutex);
694  }
695  mutex.unlock();
696  }
697  }; // end of conditional
