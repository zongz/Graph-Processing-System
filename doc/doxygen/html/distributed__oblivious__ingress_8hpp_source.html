84  /** Add an edge to the ingress object using oblivious greedy assignment. */
85  void add_edge(vertex_id_type source, vertex_id_type target,
86  const EdgeData& edata) {
87  obliv_lock.lock();
88  dht[source]; dht[target];
89  const procid_t owning_proc =
90  base_type::edge_decision.edge_to_proc_greedy(source, target, dht[source], dht[target], proc_num_edges, usehash, userecent);
91  obliv_lock.unlock();
92 
93  typedef typename base_type::edge_buffer_record edge_buffer_record;
94  edge_buffer_record record(source, target, edata);
95 #ifdef _OPENMP
96  base_type::edge_exchange.send(owning_proc, record, omp_get_thread_num());
97 #else
98  base_type::edge_exchange.send(owning_proc, record);
99 #endif
100  } // end of add edge
