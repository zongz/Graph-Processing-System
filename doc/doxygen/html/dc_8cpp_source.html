170  for (size_t i = 0; i < deletion_callbacks.size(); ++i) {
171  deletion_callbacks[i]();
172  }
173 
174  size_t bytessent = bytes_sent();
175  for (size_t i = 0;i < senders.size(); ++i) {
176  senders[i]->flush();
177  }
178 
179  comm->close();
180 
181  for (size_t i = 0;i < senders.size(); ++i) {
182  delete senders[i];
183  }
184  senders.clear();
185 
186  pthread_key_delete(dc_impl::thrlocal_sequentialization_key);
187  pthread_key_delete(dc_impl::thrlocal_send_buffer_key);
188 
189  size_t bytesreceived = bytes_received();
190  for (size_t i = 0;i < receivers.size(); ++i) {
191  receivers[i]->shutdown();
192  delete receivers[i];
193  }
194  receivers.clear();
195  // shutdown function call handlers
196  for (size_t i = 0;i < fcallqueue.size(); ++i) fcallqueue[i].stop_blocking();
197  fcallhandlers.join();
198  logstream(LOG_INFO) << "Bytes Sent: " << bytessent << std::endl;
199  logstream(LOG_INFO) << "Calls Sent: " << calls_sent() << std::endl;
200  logstream(LOG_INFO) << "Network Sent: " << network_bytes_sent() << std::endl;
201  logstream(LOG_INFO) << "Bytes Received: " << bytesreceived << std::endl;
202  logstream(LOG_INFO) << "Calls Received: " << calls_received() << std::endl;

204  delete comm;

209 void distributed_control::exec_function_call(procid_t source,
210  unsigned char packet_type_mask,
211  const char* data,
212  const size_t len) {
213  BEGIN_TRACEPOINT(dc_call_dispatch);
214  // extract the dispatch function
215  iarchive arc(data, len);
216  size_t f;
217  arc >> f;
218  // a regular funcion call
219  dc_impl::dispatch_type dispatch = (dc_impl::dispatch_type)f;
220  dispatch(*this, source, packet_type_mask, data + arc.off, len - arc.off);
221  if ((packet_type_mask & CONTROL_PACKET) == 0) inc_calls_received(source);
222  END_TRACEPOINT(dc_call_dispatch);
223 }

225 unsigned char distributed_control::get_block_sequentialization_key(fcallqueue_entry& fcallblock) {
226  unsigned char seq_key = 0;
227  char* data = fcallblock.chunk_src;
228  size_t remaininglen = fcallblock.chunk_len;
229  // loop through all the messages
230  while(remaininglen > 0) {
231  ASSERT_GE(remaininglen, sizeof(dc_impl::packet_hdr));
232  dc_impl::packet_hdr hdr = *reinterpret_cast<dc_impl::packet_hdr*>(data);
233  ASSERT_LE(hdr.len, remaininglen);
234  if (hdr.sequentialization_key != 0) {
235  seq_key = hdr.sequentialization_key;
236  break;
237  }
238  data += sizeof(dc_impl::packet_hdr) + hdr.len;
239  remaininglen -= sizeof(dc_impl::packet_hdr) + hdr.len;
240  }
241  return seq_key;
242 }

287 void distributed_control::process_fcall_block(fcallqueue_entry &fcallblock) {
288  if (fcallblock.is_chunk == false) {
289  for (size_t i = 0;i < fcallblock.calls.size(); ++i) {
290  fcallqueue_length.dec();
291  exec_function_call(fcallblock.source, fcallblock.calls[i].packet_mask,
292  fcallblock.calls[i].data, fcallblock.calls[i].len);
293  }
294  if (fcallblock.chunk_ref_counter != NULL) {
295  if (fcallblock.chunk_ref_counter->dec(fcallblock.calls.size()) == 0) {
296  delete fcallblock.chunk_ref_counter;
297  free(fcallblock.chunk_src);
298  }
299  }
300  }
301 #ifdef RPC_DO_NOT_BREAK_BLOCKS
302  else {
303  fcallqueue_length.dec();
304 
305  //parse the data in fcallblock.data
306  char* data = fcallblock.chunk_src;
307  size_t remaininglen = fcallblock.chunk_len;
308  //PERMANENT_ACCUMULATE_DIST_EVENT(eventlog, BYTES_EVENT, remaininglen);
309  while(remaininglen > 0) {
310  ASSERT_GE(remaininglen, sizeof(dc_impl::packet_hdr));
311  dc_impl::packet_hdr hdr = *reinterpret_cast<dc_impl::packet_hdr*>(data);
312  ASSERT_LE(hdr.len, remaininglen);
313 
314  if (!(hdr.packet_type_mask & CONTROL_PACKET)) {
315  global_bytes_received[hdr.src].inc(hdr.len);
316  }
317 
318  exec_function_call(fcallblock.source, hdr.packet_type_mask,
319  data + sizeof(dc_impl::packet_hdr),
320  hdr.len);
321  data += sizeof(dc_impl::packet_hdr) + hdr.len;
322  remaininglen -= sizeof(dc_impl::packet_hdr) + hdr.len;
323  }
324  free(fcallblock.chunk_src);
325  }
326 #else
327  else {
328  fcallqueue_length.dec();
329  BEGIN_TRACEPOINT(dc_receive_multiplexing);
330  fcallqueue_entry* queuebufs[fcallqueue.size()];
331  atomic<size_t>* refctr = new atomic<size_t>(0);
332 
333  fcallqueue_entry immediate_queue;
334 
335  immediate_queue.chunk_src = fcallblock.chunk_src;
336  immediate_queue.chunk_ref_counter = refctr;
337  immediate_queue.chunk_len = 0;
338  immediate_queue.source = fcallblock.source;
339  immediate_queue.is_chunk = false;
340 
341  for (size_t i = 0;i < fcallqueue.size(); ++i) {
342  queuebufs[i] = new fcallqueue_entry;
343  queuebufs[i]->chunk_src = fcallblock.chunk_src;
344  queuebufs[i]->chunk_ref_counter = refctr;
345  queuebufs[i]->chunk_len = 0;
346  queuebufs[i]->source = fcallblock.source;
347  queuebufs[i]->is_chunk = false;
348  }
349 
350  //parse the data in fcallblock.data
351  char* data = fcallblock.chunk_src;
352  size_t remaininglen = fcallblock.chunk_len;
353  //PERMANENT_ACCUMULATE_DIST_EVENT(eventlog, BYTES_EVENT, remaininglen);
354  size_t stripe = 0;
355  while(remaininglen > 0) {
356  ASSERT_GE(remaininglen, sizeof(dc_impl::packet_hdr));
357  dc_impl::packet_hdr hdr = *reinterpret_cast<dc_impl::packet_hdr*>(data);
358  ASSERT_LE(hdr.len, remaininglen);
359 
360  refctr->value++;
361 
362 
363  if ((hdr.packet_type_mask & CONTROL_PACKET)) {
364  // control calls are handled immediately with priority.
365  immediate_queue.calls.push_back(function_call_block(
366  data + sizeof(dc_impl::packet_hdr),
367  hdr.len,
368  hdr.packet_type_mask));
369  } else {
370  global_bytes_received[hdr.src].inc(hdr.len);
371  if (hdr.sequentialization_key == 0) {
372  queuebufs[stripe]->calls.push_back(function_call_block(
373  data + sizeof(dc_impl::packet_hdr),
374  hdr.len,
375  hdr.packet_type_mask));
376  ++stripe;
377  if (stripe == (fcallblock.source % fcallqueue.size())) ++stripe;
378  if (stripe >= fcallqueue.size()) stripe -= fcallqueue.size();
379  }
380  else {
381  size_t idx = (hdr.sequentialization_key % (fcallqueue.size()));
382  queuebufs[idx]->calls.push_back(function_call_block(
383  data + sizeof(dc_impl::packet_hdr),
384  hdr.len,
385  hdr.packet_type_mask));
386  }
387  }
388  data += sizeof(dc_impl::packet_hdr) + hdr.len;
389  remaininglen -= sizeof(dc_impl::packet_hdr) + hdr.len;
390  }
391  END_TRACEPOINT(dc_receive_multiplexing);
392  BEGIN_TRACEPOINT(dc_receive_queuing);
393  for (size_t i = 0;i < fcallqueue.size(); ++i) {
394  if (queuebufs[i]->calls.size() > 0) {
395  fcallqueue_length.inc(queuebufs[i]->calls.size());
396  fcallqueue[i].enqueue(queuebufs[i]);
397  }
398  else {
399  delete queuebufs[i];
400  }
401  }
402  END_TRACEPOINT(dc_receive_queuing);
403  if (immediate_queue.calls.size() > 0) process_fcall_block(immediate_queue);
404  }
405 #endif
406 }

430  size_t total_threadid) {

432  if (fcallqueue[i].empty_unsafe() == false) {

447 void distributed_control::fcallhandler_loop(size_t id) {
448  // pop an element off the queue
449 // float t = timer::approx_time_seconds();
450  fcall_handler_active[id].inc();
451  while(fcallqueue[id].is_alive()) {
452  fcallqueue[id].wait_for_data();
453  std::deque<fcallqueue_entry*> q;
454  fcallqueue[id].swap(q);
455  while (!q.empty()) {
456  fcallqueue_entry* entry;
457  entry = q.front();
458  q.pop_front();
459 
460  process_fcall_block(*entry);
461  delete entry;
462  }
463  // std::cerr << "Handler " << id << " died." << std::endl;
464  }
465  fcall_handler_active[id].dec();
466 }

487 void distributed_control::init(const std::vector<std::string> &machines,
488  const std::string &initstring,
489  procid_t curmachineid,
490  size_t numhandlerthreads,
491  dc_comm_type commtype) {
492 
493  if (numhandlerthreads == RPC_DEFAULT_NUMHANDLERTHREADS) {
494  // autoconfigure
495  if (thread::cpu_count() > 2) numhandlerthreads = thread::cpu_count() - 2;
496  else numhandlerthreads = 2;
497  }
498  ASSERT_MSG(machines.size() <= RPC_MAX_N_PROCS,
499  "Number of processes exceeded hard limit of %d", RPC_MAX_N_PROCS);
500 
501  // initialize thread local storage
502  if (dc_impl::thrlocal_sequentialization_key_initialized == false) {
503  dc_impl::thrlocal_sequentialization_key_initialized = true;
504  int err = pthread_key_create(&dc_impl::thrlocal_sequentialization_key, NULL);
505  ASSERT_EQ(err, 0);
506  }
507 
508  if (dc_impl::thrlocal_send_buffer_key_initialized == false) {
509  dc_impl::thrlocal_send_buffer_key = true;
510  int err = pthread_key_create(&dc_impl::thrlocal_send_buffer_key, dc_impl::thrlocal_send_buffer_key_deleter);
511  ASSERT_EQ(err, 0);
512  }
513 
514  //-------- Initialize the full barrier ---------
515  full_barrier_in_effect = false;
516  procs_complete.resize(machines.size());
517  //-----------------------------------------------
518 
519  // initialize the counters
520 
521  global_calls_sent.resize(machines.size());
522  global_calls_received.resize(machines.size());
523  global_bytes_received.resize(machines.size());
524  fcallqueue.resize(numhandlerthreads);
525 
526  // options
527  set_fast_track_requests(true);
528 
529  // parse the initstring
530  std::map<std::string,std::string> options = parse_options(initstring);
531 
532  if (commtype == TCP_COMM) {
533  comm = new dc_impl::dc_tcp_comm();
534  } else {
535  ASSERT_MSG(false, "Unexpected value for comm type");
536  }
537  for (procid_t i = 0; i < machines.size(); ++i) {
538  receivers.push_back(new dc_impl::dc_stream_receive(this, i));
539  senders.push_back(new dc_impl::dc_buffered_stream_send2(this, comm, i));
540  }
541  // create the handler threads
542  // store the threads in the threadgroup
543  fcall_handler_active.resize(numhandlerthreads);
544  fcall_handler_blockers.resize(numhandlerthreads);
545  fcallhandlers.set_stacksize(256*1024); // 256K
546  for (size_t i = 0;i < numhandlerthreads; ++i) {
547  fiber_control::affinity_type affinity;
548  affinity.clear();
549  affinity.set_bit(i);
550  fcallhandlers.launch(boost::bind(&distributed_control::fcallhandler_loop,
551  this, i), affinity);
552  }
553 
554 
555  // set the local proc values
556  localprocid = curmachineid;
557  localnumprocs = machines.size();
558 
559 
560  // construct the services
561  distributed_services = new dc_services(*this);
562  // start the machines
563 
564  // improves reliability of initialization
565 #ifdef HAS_MPI
566  if (mpi_tools::initialized()) MPI_Barrier(MPI_COMM_WORLD);
567 #endif
568 
569  comm->init(machines, options, curmachineid,
570  receivers, senders);
571  logstream(LOG_INFO) << "TCP Communication layer constructed." << std::endl;
572  if (localprocid == 0) {
573  logstream(LOG_EMPH) << "Cluster of " << machines.size() << " instances created." << std::endl;
574  // check for duplicate IP addresses
575  std::map<std::string, size_t> ipaddresses;
576  for (size_t i = 0; i < machines.size(); ++i ){
577  size_t pos = machines[i].find(":");
578  ASSERT_NE(pos, std::string::npos);
579  std::string address = machines[i].substr(0, pos);
580  ipaddresses[address]++;
581  }
582  bool hasduplicate = false;
583  std::map<std::string, size_t>::const_iterator iter = ipaddresses.begin();
584  while (iter != ipaddresses.end()) {
585  if (iter->second > 1) {
586  hasduplicate = true;
587  logstream(LOG_WARNING) << "Duplicate IP address: " << iter->first << std::endl;
588  }
589  ++iter;

646 /*****************************************************************************
647  Implementation of Full Barrier
648 *****************************************************************************/
649 /* It is unfortunate but this is copy paste code from dc_dist_object.hpp
650  I thought for a long time how to implement this without copy pasting and
651  I can't think of a simple enough solution.
652 
653  Part of the issue is that the "context" concept was not built into to the
654  RPC system to begin with and is currently folded in through the dc_dist_object system.
655  As a result, the global context becomes very hard to define properly.
656  Including a dc_dist_object as a member only resolves the high level contexts
657  such as barrier, broadcast, etc which do not require intrusive access into
658  deeper information about the context. The full barrier however, requires deep
659  information about the context which cannot be resolved easily.
660 */
661 
662 /**
663 This barrier ensures globally across all machines that
664 all calls issued prior to this barrier are completed before
665 returning. This function could return prematurely if
666 other threads are still issuing function calls since we
667 cannot differentiate between calls issued before the barrier
668 and calls issued while the barrier is being evaluated.
669 */
670 void distributed_control::full_barrier() {
671  // gather a sum of all the calls issued to machine 0
672  std::vector<size_t> calls_sent_to_target(numprocs(), 0);
673  for (size_t i = 0;i < numprocs(); ++i) {
674  calls_sent_to_target[i] = global_calls_sent[i].value;
675  }
676 
677  // tell node 0 how many calls there are
678  std::vector<std::vector<size_t> > all_calls_sent(numprocs());
679  all_calls_sent[procid()] = calls_sent_to_target;
680  all_gather(all_calls_sent, true);
681 
682  // get the number of calls I am supposed to receive from each machine
683  calls_to_receive.clear(); calls_to_receive.resize(numprocs(), 0);
684  for (size_t i = 0;i < numprocs(); ++i) {
685  calls_to_receive[i] += all_calls_sent[i][procid()];
686 // std::cout << "Expecting " << calls_to_receive[i] << " calls from " << i << std::endl;
687  }
688  // clear the counters
689  num_proc_recvs_incomplete.value = numprocs();
690  procs_complete.clear();
691  // activate the full barrier
692  full_barrier_in_effect = true;
693  __asm("mfence");
694  // begin one pass to set all which are already completed
695  for (procid_t i = 0;i < numprocs(); ++i) {
696  if (global_calls_received[i].value >= calls_to_receive[i]) {
697  if (procs_complete.set_bit(i) == false) {
698  num_proc_recvs_incomplete.dec();
699  }
700  }
701  }
702 
703  full_barrier_lock.lock();
704  while (num_proc_recvs_incomplete.value > 0) full_barrier_cond.wait(full_barrier_lock);
705  full_barrier_lock.unlock();
706  full_barrier_in_effect = false;
707  barrier();
708 // for (size_t i = 0; i < numprocs(); ++i) {
709 // std::cout << "Received " << global_calls_received[i].value << " from " << i << std::endl;
710 // }
711 }
