64 void fifo_scheduler::schedule(const lvid_type vid, double priority) {
65  if (vid < num_vertices && !vertex_is_scheduled.set_bit(vid)) {
66  /* "Randomize" the task queue task is put in. Note that we do
67  not care if this counter is corrupted in race conditions
68  Find first queue that is not locked and put task there (or
69  after iteration limit) Choose two random queues and use the
70  one which has smaller size */
71  // M.D. Mitzenmacher The Power of Two Choices in Randomized
72  // Load Balancing (1991)
73  // http://www.eecs.harvard.edu/~michaelm/postscripts/mythesis.
74  size_t idx = 0;
75  if(queues.size() > 1) {
76  const uint32_t prod =
77  random::fast_uniform(uint32_t(0),
78  uint32_t(queues.size() * queues.size() - 1));
79  const uint32_t r1 = prod / queues.size();
80  const uint32_t r2 = prod % queues.size();
81  idx = (queues[r1].size() < queues[r2].size()) ? r1 : r2;
82  }
83  locks[idx].lock(); queues[idx].push_back(vid); locks[idx].unlock();
84  }
85 }

87 /** Get the next element in the queue */
88 sched_status::status_enum fifo_scheduler::get_next(const size_t cpuid,
89  lvid_type& ret_vid) {
90  /* Check all of my queues for a task */
91  // begin scanning from the machine's current queue
92  size_t initial_idx = (current_queue[cpuid] % multi) + cpuid * multi;
93  for(size_t i = 0; i < queues.size(); ++i) {
94  const size_t idx = (initial_idx + i) % queues.size();
95  // increment the current queue as long as I am scanning with in the
96  // queues owned by this machine
97  current_queue[cpuid] += (i < multi);
98 
99  // pick up the lock
100  bool good = false;
101  locks[idx].lock();
102  while(!queues[idx].empty()) {
103  // not empty, pop and verify
104  ret_vid = queues[idx].front();
105  queues[idx].pop_front();
106  if (ret_vid < num_vertices) {
107  good = vertex_is_scheduled.clear_bit(ret_vid);
108  if (good) break;
109  }
110  }
111  locks[idx].unlock();
112  // managed to retrieve a task
113  if(good) {
114  return sched_status::NEW_TASK;
115  }
116  }
117  return sched_status::EMPTY;
118 } // end of get_next_task
